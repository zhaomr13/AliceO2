79 // When the current time is forwarded (every 100 ns) the entry 0 of the future dequeue
80 // is removed and pushed at the end of the past dequeue. At the same time a new entry in
81 // the future dequeue is pushed at the end, and - in case the past dequeue reached 15 entries
82 // the first entry of the past dequeue is removed.
83 void DigitsWriteoutBuffer::forwardMarker(o2::InteractionTimeRecord record)
84 {
85 
86  unsigned long eventTime = record.getTimeNS();
87 
88  // To see how much the marker will forwarded, or see how much of the future buffer will be written into the past past buffer
89  int sampleDifference = (eventTime - mTriggerTime) / 100 - (mLastEventTime - mTriggerTime) / 100;
90 
91  for (int idel = 0; idel < sampleDifference; idel++) {
92 
93  // Stop reading if record mode is false to save memory
94  if (!mTimedDigitsFuture.front().mRecordMode) {
95  break;
96  }
97 
98  // with sampleDifference, the future buffer will written into the past buffer
99  // the added entries will be removed the future, and the same number will added as empty bins
100  mTimedDigitsPast.push_back(mTimedDigitsFuture.front());
101  mTimedDigitsFuture.pop_front();
102  mTimedDigitsFuture.push_back(o2::emcal::DigitTimebin());
103 
104  if (mTimedDigitsPast.size() > mBufferSize) {
105  mTimedDigitsPast.pop_front();
106  }
107 
108  // If it is the end of the readout window write all the digits, labels, and trigger record into the streamer
109  if (mTimedDigitsPast.back().mEndWindow) {
110 
111  // Find the trigger time bin
112  o2::emcal::DigitTimebin triggerNode;
113  for (auto& digitTimebin : mTimedDigitsPast) {
114  if (digitTimebin.mTriggerColl) {
115  triggerNode = digitTimebin;
116  break;
117  }
118  }
119  mDigitStream.fill(getLastSamples(), triggerNode.mInterRecord.value());
120  clear();
121  }
122  }
123 
124  // If we have a trigger, all the time bins in the future buffer will be set to record mode
125  // the last time bin will the end of the readout window since it will be mTriggerTime + 1500 ns
126  if ((eventTime - mTriggerTime) >= (mLiveTime + mBusyTime) || mFirstEvent) {
127  mTriggerTime = eventTime;
128  mTimedDigitsFuture.front().mTriggerColl = true;
129  mTimedDigitsFuture.front().mInterRecord = record;
130  mTimedDigitsFuture[(mLiveTime / 100) - 1].mEndWindow = true;
131 
132  long timeStamp = (eventTime / 100) * 100;
133  for (auto& iNode : mTimedDigitsFuture) {
134  long diff = (timeStamp - eventTime);
135  if (TMath::Abs(diff) > mLiveTime) {
136  break;
137  }
138  iNode.mRecordMode = true;
139  timeStamp += 100;
140  }
141  }
142 
143  // If we have a pre-trigger collision, all the time bins in the future buffer will be set to record mode
144  if ((eventTime - mTriggerTime) >= (mLiveTime + mBusyTime - mPreTriggerTime)) {
145  long timeStamp = (eventTime / 100) * 100;
146  for (auto& iNode : mTimedDigitsFuture) {
147  long diff = (timeStamp - eventTime);
148  if (TMath::Abs(diff) > mLiveTime) {
149  break;
150  }
151  iNode.mRecordMode = true;
152  timeStamp += 100;
153  }
154  }
155 
156  mLastEventTime = eventTime;
157  mPhase = ((int)(std::fmod(mLastEventTime, 100) / 25));
158  if (mFirstEvent) {
159  mFirstEvent = false;
160  }
161 }

164 void DigitsWriteoutBuffer::finish()
165 {
166  for (unsigned int ibin = 0; ibin < mBufferSize; ibin++) {
167 
168  if (!mTimedDigitsFuture.front().mRecordMode || mTimedDigitsFuture.front().mDigitMap->empty()) {
169  break;
170  }
171 
172  mTimedDigitsPast.push_back(mTimedDigitsFuture.front());
173  mTimedDigitsFuture.pop_front();
174 
175  if (mTimedDigitsPast.size() > mBufferSize) {
176  mTimedDigitsPast.pop_front();
177  }
178 
179  if (mTimedDigitsPast.back().mEndWindow) {
180 
181  // Find the trigger time bin
182  o2::emcal::DigitTimebin triggerNode;
183  for (auto& digitTimebin : mTimedDigitsPast) {
184  if (digitTimebin.mTriggerColl) {
185  triggerNode = digitTimebin;
186  break;
187  }
188  }
189 
190  mDigitStream.fill(getLastSamples(), triggerNode.mInterRecord.value());
191  clear();
192  break;
193  }
194  }
195 }

197 gsl::span<o2::emcal::DigitTimebin> DigitsWriteoutBuffer::getLastSamples()
198 {
199 
200  // Setting the digit time
201  // If we have a delay the first digit in the buffer will start from mDelay,
202  // If we also have digits coming from pre-trigger collisions, also their time will start from mDelay,
203  // so, to shift the digits back to zero, their time has to be subtracted by the extra digits (with time [0,mDelay])
204  int timeStamp = mLiveTime - (mTimedDigitsPast.size() * 100);
205  for (auto& buffEntry : mTimedDigitsPast) {
206  for (auto& [tower, digitList] : *buffEntry.mDigitMap) {
207  for (auto& digit : digitList) {
208  digit.setTimeStamp(digit.getTimeStamp() + timeStamp);
209  }
210  }
211  timeStamp += 100;
212  }
213 
214  return gsl::span<o2::emcal::DigitTimebin>(&mTimedDigitsPast[0], mTimedDigitsPast.size());
215 }
