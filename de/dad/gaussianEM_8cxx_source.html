48 void computeDiscretizedGaussian2D(const double* xyInfSup, const double* theta,
49  int K, int N, int k, double* z)
50 {
51  // compute the k-component of the mixture
52  const double* xInf = getConstXInf(xyInfSup, N);
53  const double* yInf = getConstYInf(xyInfSup, N);
54  const double* xSup = getConstXSup(xyInfSup, N);
55  const double* ySup = getConstYSup(xyInfSup, N);
56 
57  const double* varX = getConstVarX(theta, K);
58  const double* varY = getConstVarY(theta, K);
59  const double* muX = getConstMuX(theta, K);
60  const double* muY = getConstMuY(theta, K);
61  // const double *w = getConstW(theta);;
62 
63  // Compute
64  /*
65  ISupSup[:] = 0.25*( 1.0 + vErf( (xsup[:]-muX) * cstx ) ) * ( 1.0 + vErf( (ysup[:]-muY) * csty ) )
66  IInfInf[:] = 0.25*( 1.0 + vErf( (xinf[:]-muX) * cstx ) ) * ( 1.0 + vErf( (yinf[:]-muY) * csty ) )
67  IInfSup[:] = 0.25*( 1.0 + vErf( (xinf[:]-muX) * cstx ) ) * ( 1.0 + vErf( (ysup[:]-muY) * csty ) )
68  ISupInf[:] = 0.25*( 1.0 + vErf( (xsup[:]-muX) * cstx ) ) * ( 1.0 + vErf( (yinf[:]-muY) * csty ) )
69  */
70  double IxInf[N];
71  double IyInf[N];
72  double IxSup[N];
73  double IySup[N];
74  double sigX, sigY;
75  double cstx, csty;
76 
77  if (varX[k] < 1.e-15) {
78  for (int i = 0; i < N; i++) {
79  IxInf[i] = ((xInf[i] - muX[k]) < 0.0) ? 0.0 : 2.0;
80  IxSup[i] = ((xSup[i] - muX[k]) < 0.0) ? 0.0 : 2.0;
81  }
82  } else {
83  sigX = sqrt(varX[k]);
84  cstx = 1.0 / (sqrt(2.0) * sigX);
85  for (int i = 0; i < N; i++) {
86  IxInf[i] = 1.0 + erf((xInf[i] - muX[k]) * cstx);
87  IxSup[i] = 1.0 + erf((xSup[i] - muX[k]) * cstx);
88  }
89  }
90  if (varY[k] < 1.e-15) {
91  for (int i = 0; i < N; i++) {
92  IyInf[i] = ((yInf[i] - muY[k]) < 0.0) ? 0.0 : 2.0;
93  IySup[i] = ((ySup[i] - muY[k]) < 0.0) ? 0.0 : 2.0;
94  }
95  } else {
96  sigY = sqrt(varY[k]);
97  csty = 1.0 / (sqrt(2.0) * sigY);
98  for (int i = 0; i < N; i++) {
99  IyInf[i] = 1.0 + erf((yInf[i] - muY[k]) * csty);
100  IySup[i] = 1.0 + erf((ySup[i] - muY[k]) * csty);
101  }
102  }
103 
104  // Compute
105  //
106  // ISupSup[:] = 0.25 * IxSup[:] * IySup[:]
107  // IInfInf[:] = 0.25 * IXInf[:] * IyInf[:]
108  // IInfSup[:] = 0.25 * IXInf[:] * IYSup[:]
109  // ISupInf[:] = 0.25 * IXSup[:] * IYInf[:]
110  // z[:] = ISupSup[:] - IInfSup[:] - ISupInf[:] + IInfInf[:];
111 
112  for (int i = 0; i < N; i++) {
113  z[i] = 0.25 * IxSup[i] * IySup[i];
114  z[i] += 0.25 * IxInf[i] * IyInf[i];
115  z[i] -= 0.25 * IxInf[i] * IySup[i];
116  z[i] -= 0.25 * IxSup[i] * IyInf[i];
117  }
118  return;
119 }

121 void generateMixedGaussians2D(const double* xyInfSup, const double* theta, int K, int N, double* z)
122 {
123  double zk[N];
124  const double* w = getConstW(theta, K);
125  vectorSetZero(z, N);
126  for (int k = 0; k < K; k++) {
127  if (w[k] > 0.0) {
128  computeDiscretizedGaussian2D(xyInfSup, theta, K, N, k, zk);
129  // z[i] = w[k] * zk[i]
130  vectorAddVector(z, w[k], zk, N, z);
131  }
132  }

135 double computeWeightedLogLikelihood(const double* xyInfSup, const double* theta, const double* z, int K, int N)
136 {
137  // w is assumed to be normalized
138  double li[N];
139  double LogLikelihood = 0;
140 
141  generateMixedGaussians2D(xyInfSup, theta, K, N, li);
142  // Avoiding computing log( small values ) replace by 0.0 (meaning no contribution)
143  for (int i = 0; i < N; i++) {
144  // TODO: avoid writing in li
145  li[i] = (li[i] < DBL_EPSILON) ? 0.0 : log(li[i]) * z[i];
146  LogLikelihood += li[i];
147  }
148  return LogLikelihood;
149 }

151 void EStep(const double* xyInfSup, const double* theta, int K, int N, int verbose, double* eta, double* zEval)
152 {
153  /*
154  Compute new eta(i,k), fraction/proba that point i
155  belongs to kth gaussian
156 */
157  double kSum[N];
158  // ??? To remove : double zEval[N];
159  vectorSetZero(eta, K * N);
160  vectorSetZero(kSum, N);
161  const double* w = getConstW(theta, K);
162 
163  for (int k = 0; k < K; k++) {
164  // eta[:, k] = w[k] * computeDiscretizedGaussian2D( xy[:,:], dxy, mu[k,:], var[k,:] )
165  // kSum += eta[:,k]
166  if (w[k] != 0.0) {
167  computeDiscretizedGaussian2D(xyInfSup, theta, K, N, k, zEval);
168  vectorMultScalar(zEval, w[k], N, &eta[k * N]);
169  vectorAddVector(kSum, 1.0, &eta[k * N], N, kSum);
170  }
171  }
172  // eta normalisation
173  for (int i = 0; i < N; i++) {
174  if (kSum[i] < DBL_EPSILON) {
175  // print( "WARNING sum eta(i,:) is null, i=",i , "xy=", xy[:,i], "dxy=",dxy[:,i])
176  // eta[ i, :] = eta[i, :] / kSum[i]
177  for (int k = 0; k < K; k++) {
178  eta[k * N + i] = 0.;
179  }
180  } else {
181  // eta[ i, :] = eta[i, :] / kSum[i]
182  for (int k = 0; k < K; k++) {
183  eta[k * N + i] = eta[k * N + i] / kSum[i];
184  }
185  }
186  }
187  if (verbose >= 2) {
188  double nSum[K];
189  vectorSetZero(kSum, N);
190  vectorSetZero(nSum, K);
191  for (int i = 0; i < N; i++) {
192  for (int k = 0; k < K; k++) {
193  kSum[i] += eta[k * N + i];
194  nSum[k] += eta[k * N + i];
195  }
196  }
197  printf(" E-Step check: sum_k{eta}[0:N] min=%f max=%f\n", vectorMin(kSum, N), vectorMax(kSum, N));
198  printf(" E-Step check: sum_n{eta}[0:K] min=%f max=%f\n", vectorMin(nSum, K), vectorMax(nSum, K));
199  printf(" E-Step check: sum_{eta} =%f\n", vectorSum(nSum, K));
200  vectorPrint(" E-Step check: sum_k{eta}[0:N]", kSum, N);
201  }
202  return;
203 }

260 void weightedMStep(const double* xyDxy, const double* z, const double* eta, int K, int N, int cstVar, double* theta)
261 {
262  // xyDxy[N,4]
263  // z[N]
264  // eta[K, N]
265 
266  const double* x = getConstX(xyDxy, N);
267  const double* y = getConstY(xyDxy, N);
268 
269  double* varX = getVarX(theta, K);
270  double* varY = getVarY(theta, K);
271  double* muX = getMuX(theta, K);
272  double* muY = getMuY(theta, K);
273  double* w = getW(theta, K);
274 
275  // Compute unnormalized wk
276  // w[k] = sum_i{ eta[ k, i] * z[i] }
277  double wk[K];
278  gsl_vector_const_view gsl_z = gsl_vector_const_view_array(z, N);
279  gsl_vector_view gsl_wk = gsl_vector_view_array(wk, K);
280  gsl_matrix_const_view gsl_eta = gsl_matrix_const_view_array(eta, K, N);
281  gsl_blas_dgemv(CblasNoTrans, 1.0, &gsl_eta.matrix, &gsl_z.vector, 0.0, &gsl_wk.vector);
282 
283  // vectorPrint( "??? MStep ", wk, K);
284 
285  double u[N];
286  gsl_vector_view gsl_u = gsl_vector_view_array(u, N);
287  gsl_vector_view gsl_muX = gsl_vector_view_array(muX, K);
288  gsl_vector_view gsl_muY = gsl_vector_view_array(muY, K);
289  //
290  // Compute muX
291  //
292  // u[i] = x[i]*z[i]
293  vectorMultVector(x, z, N, u);
294  // muX[k] = sum_i{ eta[ k, i] * x[i] * z[i] }
295  gsl_blas_dgemv(CblasNoTrans, 1.0, &gsl_eta.matrix, &gsl_u.vector, 0.0, &gsl_muX.vector);
296  //
297  // Compute muY
298  //
299  // u[i] = x[i]*z[i]
300  vectorMultVector(y, z, N, u);
301  // muX[k] = sum_i{ eta[ k, i] * y[i] * z[i] }
302  gsl_blas_dgemv(CblasNoTrans, 1.0, &gsl_eta.matrix, &gsl_u.vector, 0.0, &gsl_muY.vector);
303  //
304  // Normalization muX, muY
305  // Wk[k] = 0, the component is meaningless
306  for (int k = 0; k < K; k++) {
307  muX[k] = (w[k] < DBL_EPSILON) ? 0.0 : muX[k] / wk[k];
308  muY[k] = (w[k] < DBL_EPSILON) ? 0.0 : muY[k] / wk[k];
309  }
310 
311  // Compute varX, varY
312  if (cstVar == 0) {
313  double x2[N], y2[N];
314  gsl_vector_view gsl_x2 = gsl_vector_view_array(x2, N);
315  gsl_vector_view gsl_y2 = gsl_vector_view_array(y2, N);
316  for (int k = 0; k < K; k++) {
317  for (int i = 0; i < N; i++) {
318  x2[i] = x[i] - muX[k];
319  x2[i] = x2[i] * x2[i];
320  x2[i] = x2[i] * z[i];
321  y2[i] = y[i] - muY[k];
322  y2[i] = y2[i] * y2[i];
323  y2[i] = y2[i] * z[i];
324  }
325  gsl_vector_const_view gsl_etaK = gsl_vector_const_view_array(&eta[k * N], N);
326  gsl_blas_ddot(&gsl_etaK.vector, &gsl_x2.vector, &varX[k]);
327  gsl_blas_ddot(&gsl_etaK.vector, &gsl_y2.vector, &varY[k]);
328  // varX, varY normalization
329  if (wk[k] < DBL_EPSILON) {
330  varX[k] = 0.0;
331  varY[k] = 0.0;
332  } else {
333  varX[k] = varX[k] / wk[k];
334  varY[k] = varY[k] / wk[k];
335  // To avoid numerical pb
336  varX[k] = (varX[k] < DBL_EPSILON) ? 0.0 : varX[k];
337  varY[k] = (varY[k] < DBL_EPSILON) ? 0.0 : varY[k];
338  }
339  }
340  }
341  //
342  // Calculate w (normalization)
343  // w = wk / N
344  //
345  // Build the mask from eta
346  // because of the points which don't
347  // participate eta[k, :] = 0
348  double etaN[N];
349  vectorSetZero(etaN, N);
350  for (int k = 0; k < K; k++) {
351  for (int i = 0; i < N; i++) {
352  etaN[i] += eta[k * N + i];
353  }
354  }
355  double zMasked[N];
356  vectorMultVector(z, etaN, N, zMasked);
357  double sum = vectorSum(zMasked, N);
358  // verbose ???
359  if (0) {
360  vectorPrint("MStep z", z, N);
361  vectorPrint("MStep wk", wk, K);
362  }
363  vectorMultScalar(wk, 1.0 / sum, K, w);
364  return;
365 }

367 double weightedEMLoop(const double* xyDxy, const Mask_t* saturated, const double* zObs,
368  const double* theta0, const Mask_t* maskTheta, int K, int N,
369  int mode, double LConvergence, int verbose, double* theta)
370 {
371 
372  // Mode of computation
373  // TODO make a function
374  int m = mode;
375  int cstVarMode = m & 0x1;
376  // Saturated
377  // Not selected way ??? int nbrSaturatedPads = (m >> 1) & 0x1;
378  int nbrSaturatedPads = vectorSumShort(saturated, N);
379  if (verbose >= 2) {
380  printf(" wheightedEMLoop : cstVarMode =%d\n", cstVarMode);
381  printf(" wheightedEMLoop : nbrSaturatedPads =%d\n", nbrSaturatedPads);
382  }
383  // vectorPrint(" zObs", zObs, N);
384  // vectorPrint(" xydxy", xyDxy, N*4);
385  // Define theta
386  vectorCopy(theta0, K * 5, theta);
387  double* varX = getVarX(theta, K);
388  double* varY = getVarY(theta, K);
389  double* muX = getMuX(theta, K);
390  double* muY = getMuY(theta, K);
391  double* w = getW(theta, K);
392 
393  // Define eta
394  double eta[N * K];
395 
396  // define x, y, dx, dy description
397  const double* x = getConstX(xyDxy, N);
398  const double* y = getConstY(xyDxy, N);
399  const double* dX = getConstDX(xyDxy, N);
400  const double* dY = getConstDY(xyDxy, N);
401 
402  // Compute boundary of each pads
403  double xyInfSup[4 * N];
404  vectorAddVector(x, -1.0, dX, N, getXInf(xyInfSup, N));
405  vectorAddVector(y, -1.0, dY, N, getYInf(xyInfSup, N));
406  vectorAddVector(x, +1.0, dX, N, getXSup(xyInfSup, N));
407  vectorAddVector(y, +1.0, dY, N, getYSup(xyInfSup, N));
408  //
409  // z normalization
410  double zSum = vectorSum(zObs, N);
411  double zObsNorm[N];
412  vectorMultScalar(zObs, 1.0 / zSum, N, zObsNorm);
413  //
414  // Initial Likelihood
415  if (maskTheta) {
416  for (int k = 0; k < K; k++) {
417  w[k] = (maskTheta[k] == 1) ? w[k] : 0;
418  }
419  }
420  double logL = computeWeightedLogLikelihood(xyInfSup, theta0, zObsNorm, K, N);
421  if (verbose >= 1) {
422  printEMState(-1, logL, 0.0);
423  }
424  //
425  // EM Loop
426  //
427  double zEval[N];
428  double prevLogL = logL;
429  logL = 1.0 + prevLogL;
430  int it = 0;
431  // for( ; (( fabs((logL - prevLogL)/logL) > LConvergence) || (it < nIterMin)) && ( it < nIterMax ); ) {
432  for (; ((fabs((logL - prevLogL) / logL) > LConvergence)) && (it < nIterMax);) {
433  if (verbose >= 1) {
434  printEMState(it, logL, logL - prevLogL);
435  }
436  prevLogL = logL;
437  //
438  // EM Step
439  //
440  // E-Step
441  if (maskTheta == nullptr) {
442  EStep(xyInfSup, theta, K, N, verbose, eta, zEval);
443  } else {
444  maskedEStep(xyInfSup, theta, maskTheta, K, N, verbose, eta, zEval);
445  //
446  }
447  if (nbrSaturatedPads > 0) {
448  // Set (or update) saturated pads to the estimate
449  // values i.e. zEval
450  // Remark
451  // saturated is a constant array and
452  // the zObsNorm[ saturated == 0 )] are not modified
453  // and take the zObsNorm Values
454  vectorMaskedUpdate(saturated, zEval, N, zObsNorm);
455  }
456  // M-Step
457  weightedMStep(xyDxy, zObsNorm, eta, K, N, cstVarMode, theta);
458  // Log-Lilelihood
459  logL = computeWeightedLogLikelihood(xyInfSup, theta, zObsNorm, K, N);
460 
461  if (verbose >= 2) {
462  printTheta(" EM new theta", theta, K);
463  }
464  it += 1;
465  }
466  if (verbose >= 1) {
467  printEMState(it, logL, logL - prevLogL);
468  }
469  if (verbose >= 2) {
470  printf("End GaussianEM\n");
471  }
472  // Return BIC criterion
473  int kSignificant = vectorSumOfGreater(w, 10.e-5, K);
474  printf("EMLoop # parameters %d, log( N -saturated)= %f \n", (3 * kSignificant - 1), log(N - nbrSaturatedPads));
475  printf("EMLoop LogL=%f\n", logL);
476  double dof = (3 * kSignificant - 1);
477  double nData = N - nbrSaturatedPads;
478  double BIC = -2 * logL + (3 * kSignificant - 1) * log(N - nbrSaturatedPads);
479  double AIC = 2 * (3 * kSignificant - 1) - 2 * logL;
480  double AICc = 2 * (3 * kSignificant - 1) - 2 * logL + 2 * dof * (dof + 1) / (nData - dof - 1);
481  double zPredict[N];
482  generateMixedGaussians2D(xyInfSup, theta, K, N, zPredict);
483  double crossE = crossEntropy(zObs, zPredict, N);
484  printf("EMLoop BIC=%f, AIC=%f, AICc=%f crossEntropy=%f\n", BIC, AIC, AICc, crossE);
485 
486  return logL;
487 }
