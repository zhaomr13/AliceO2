36 void Digitizer::init()
37 {
38  mSimParam = &(o2::emcal::SimParam::Instance());
39  mRandomGenerator = new TRandom3(std::chrono::high_resolution_clock::now().time_since_epoch().count());
40 
41  float tau = mSimParam->getTimeResponseTau();
42  float N = mSimParam->getTimeResponsePower();
43  float delay = std::fmod(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE, 1);
44  mDelay = ((int)(std::floor(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE)));
45  mTimeWindowStart = ((unsigned int)(std::floor(mSimParam->getTimeBinOffset() / constants::EMCAL_TIMESAMPLE)));
46 
47  mSmearEnergy = mSimParam->doSmearEnergy();
48  mSimulateTimeResponse = mSimParam->doSimulateTimeResponse();
49 
50  mDigits.init();
51  if ((mDelay - mTimeWindowStart) != 0) {
52  mDigits.setBufferSize(mDigits.getBufferSize() + (mDelay - mTimeWindowStart));
53  mDigits.reserve();
54  }
55 
56  mAmplitudeInTimeBins.clear();
57 
58  // for each phase create a template distribution
59  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
60  RawResponse.SetParameters(1., 0., tau, N, 0.);
61 
62  for (int i = 0; i < 4; i++) {
63 
64  std::vector<double> sf;
65  RawResponse.SetParameter(1, 0.25 * i);
66 
67  for (int j = 0; j < constants::EMCAL_MAXTIMEBINS; j++) {
68  sf.push_back(RawResponse.Eval(j - mTimeWindowStart));
69  }
70  mAmplitudeInTimeBins.push_back(sf);
71  }
72 }

75 double Digitizer::rawResponseFunction(double* x, double* par)
76 {
77  double signal = 0.;
78  double tau = par[2];
79  double n = par[3];
80  double ped = par[4];
81  double xx = (x[0] - par[1] + tau) / tau;
82 
83  // par[0] amp, par[1] peak time
84 
85  if (xx <= 0) {
86  signal = ped;
87  } else {
88  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
89  }
90 
91  return signal;
92 }

101 void Digitizer::process(const std::vector<LabeledDigit>& labeledSDigits)
102 {
103 
104  for (auto labeleddigit : labeledSDigits) {
105 
106  int tower = labeleddigit.getTower();
107 
108  sampleSDigit(labeleddigit.getDigit());
109 
110  if (mTempDigitVector.size() == 0) {
111  continue;
112  }
113 
114  std::vector<LabeledDigit> listofLabeledDigit;
115 
116  for (auto digit : mTempDigitVector) {
117  Int_t id = digit.getTower();
118 
119  auto labels = labeleddigit.getLabels();
120  LabeledDigit d(digit, labels[0]);
121  for (auto label : labels) {
122  if (digit.getAmplitude() == 0) {
123  label.setAmplitudeFraction(0);
124  }
125  if (label == labels.front()) {
126  continue;
127  }
128  d.addLabel(label);
129  }
130  listofLabeledDigit.push_back(d);
131  }
132  mDigits.addDigits(tower, listofLabeledDigit);
133  }
134 }

137 void Digitizer::sampleSDigit(const Digit& sDigit)
138 {
139  mTempDigitVector.clear();
140  Int_t tower = sDigit.getTower();
141  Double_t energy = sDigit.getAmplitude();
142 
143  if (mSmearEnergy) {
144  energy = smearEnergy(energy);
145  }
146 
147  if (energy < __DBL_EPSILON__) {
148  return;
149  }
150 
151  if (mSimulateTimeResponse) {
152  for (int j = 0; j < mAmplitudeInTimeBins.at(mPhase).size(); j++) {
153 
154  double val = energy * (mAmplitudeInTimeBins.at(mPhase).at(j));
155  double digitTime = (mEventTimeOffset + mDelay - mTimeWindowStart) * constants::EMCAL_TIMESAMPLE;
156 
157  Digit digit(tower, val, digitTime);
158  mTempDigitVector.push_back(digit);
159  }
160  } else {
161  Digit digit(tower, energy, (mDelay - mTimeWindowStart) * constants::EMCAL_TIMESAMPLE);
162  mTempDigitVector.push_back(digit);
163  }
164 }

167 double Digitizer::smearEnergy(double energy)
168 {
169  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
170  energy *= mRandomGenerator->Poisson(fluct) / fluct;
171  return energy;
172 }
