310 void DigiReco::prepareInterpolation()
311 {
312  // Prepare tapered sinc function
313  // Lost reference of first interpolating function
314  // Found problems in implementation: the interpolated function is not derivable in sampled points
315  // tsc/TSN =3.75 (~ 4) and TSL*TSN*sqrt(2)/tsc >> 1 (n. of sigma)
316  // const O2_ZDC_DIGIRECO_FLT tsc = 750;
317  // Now using Kaiser function
318  O2_ZDC_DIGIRECO_FLT beta = TMath::Pi() * mAlpha;
319  O2_ZDC_DIGIRECO_FLT norm = 1. / TMath::BesselI0(beta);
320  constexpr int n = TSL * TSN;
321  for (int tsi = 0; tsi <= n; tsi++) {
322  O2_ZDC_DIGIRECO_FLT arg1 = TMath::Pi() * O2_ZDC_DIGIRECO_FLT(tsi) / O2_ZDC_DIGIRECO_FLT(TSN);
323  O2_ZDC_DIGIRECO_FLT fs = 1;
324  if (arg1 != 0) {
325  fs = TMath::Sin(arg1) / arg1;
326  }
327  // First tapering window
328  // O2_ZDC_DIGIRECO_FLT arg2 = O2_ZDC_DIGIRECO_FLT(tsi) / tsc;
329  // O2_ZDC_DIGIRECO_FLT fg = TMath::Exp(-arg2 * arg2);
330  // Kaiser window
331  O2_ZDC_DIGIRECO_FLT arg2 = O2_ZDC_DIGIRECO_FLT(tsi) / O2_ZDC_DIGIRECO_FLT(n);
332  O2_ZDC_DIGIRECO_FLT fg = norm * TMath::BesselI0(beta * TMath::Sqrt(1. - arg2 * arg2));
333  mTS[n + tsi] = fs * fg;
334  mTS[n - tsi] = mTS[n + tsi]; // Function is even
335  }
336  LOG(info) << "Interpolation numeric precision is " << sizeof(O2_ZDC_DIGIRECO_FLT);
337  LOG(info) << "Interpolation alpha = " << mAlpha;
338 }

481  // Last bunch

484  seq_beg = mNBC;
485  seq_end = mNBC;
486  } else {
487  // Look for another bunch
488  seq_end = ibc;
489  }
490  }

494 void DigiReco::lowPassFilter()
495 {
496  // First attempt to low pass filtering uses the average of three consecutive samples
497  // ringing noise has T~6 ns w.r.t. a sampling period of ~ 2 ns
498  // one should get smoothing of the noise
499 #ifdef O2_ZDC_DEBUG
500  LOG(info) << "________________________________________________________________________________";
501  LOG(info) << __func__;
502 #endif
503  constexpr int MaxTimeBin = NTimeBinsPerBC - 1;
504  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
505  auto isig = TDCSignal[itdc];
506  for (int ibc = 0; ibc < mNBC; ibc++) {
507  auto ref_c = mReco[ibc].ref[isig];
508  uint32_t ref_p = ibc == 0 ? ZDCRefInitVal : mReco[ibc - 1].ref[isig];
509  uint32_t ref_n = ibc == (mNBC - 1) ? ZDCRefInitVal : mReco[ibc + 1].ref[isig];
510  if (ref_c != ZDCRefInitVal) {
511  for (int is = 0; is < NTimeBinsPerBC; is++) {
512  int32_t sum = mChData[ref_c].data[is];
513  if (is == 0) {
514  sum += mChData[ref_c].data[1];
515  if (ref_p != ZDCRefInitVal) {
516  // b.c. number of (ibc) - b.c. number (ibc-1)
517  auto bcd = mReco[ibc].ir.differenceInBC(mReco[ibc - 1].ir);
518  if (bcd == 1) { // Previous bunch crossing
519  sum += mChData[ref_p].data[MaxTimeBin];
520  }
521  } else {
522  // As a backup we count twice the first sample
523  sum += mChData[ref_c].data[0];
524  }
525  } else if (is == MaxTimeBin) {
526  sum += mChData[ref_c].data[MaxTimeBin - 1];
527  if (ref_n != ZDCRefInitVal) {
528  // b.c. number of (ibc+1) - b.c. number (ibc)
529  auto bcd = mReco[ibc + 1].ir.differenceInBC(mReco[ibc].ir);
530  if (bcd == 1) {
531  sum += mChData[ref_n].data[0];
532  }
533  } else {
534  // As a backup we count twice the last sample
535  sum += mChData[ref_c].data[MaxTimeBin];
536  }
537  } else {
538  // Not on the bunch boundary
539  sum += mChData[ref_c].data[is - 1];
540  sum += mChData[ref_c].data[is + 1];
541  }
542  // Make the average taking into account rounding and sign
543  bool isNegative = sum < 0;
544  if (isNegative) {
545  sum = -sum;
546  }
547  auto mod = sum % 3;
548  sum = sum / 3;
549  if (mod == 2) {
550  sum++;
551  }
552  if (isNegative) {
553  sum = -sum;
554  }
555  // Store filtered values
556  mReco[ibc].data[isig][is] = sum;
557  }
558  }
559  }

656  for (int ibp = 1; ibp < 4; ibp++) {

660  break;

663  if (bcd < 1) {

666  if (bcd > 3) {
667  break;

678  // chfired: Fired TDC condition related to channel (e.g. AND of TC and SUM @ main-main)
679  hasFired[bcd] = mReco[ibun].chfired[ich];

681  // but it may extend outside the correct bunch for signals near the bunch edges
682  // hasHit refers to a single channel since it is derived from ChannelDataV0::Hit
683  hasHit[bcd] = mBCData[ibun].triggers & mChMask[ich];

687  mt.w = mBCData[ibun].moduleTriggers[mRopt->amod[ich]];

689  hasAutoM[bcd] = mt.f.Auto_m;

701  ref[0] = mReco[ibun].ref[ich];
702  hasHit[0] = mBCData[ibun].triggers & mChMask[ich];
703  ModuleTriggerMapData mt;
704  mt.w = mBCData[ibun].moduleTriggers[mRopt->amod[ich]];
705  hasAuto0[0] = mt.f.Auto_0;
706  hasAutoM[0] = mt.f.Auto_m;
707  if (rec.chfired[ich]) {
708  // Check if channel data are present in payload
709  if (ref[0] < ZDCRefInitVal) {

711  // Compute event by event pedestal
712  bool hasEvPed = false;
713  float evPed = 0;
714  if (ibun > ibeg) {

716  if (mRopt->beg_ped_int[ich] >= 0 || ref_m < ZDCRefInitVal) {
717  for (int is = mRopt->beg_ped_int[ich]; is <= mRopt->end_ped_int[ich]; is++) {
718  if (is < 0) {
719  // Sample is in previous BC
720  evPed += float(mChData[ref_m].data[is + NTimeBinsPerBC]);
721  } else {
722  // Sample is in current BC
723  evPed += float(mChData[ref[0]].data[is]);
724  }
725  }
726  evPed /= float(mRopt->end_ped_int[ich] - mRopt->beg_ped_int[ich] + 1);
727  hasEvPed = true;
728  }
729  }
730  // Pile-up detection using trigger information allows to identify
731  // the presence of a signal in previous bunch and is module-wise
732 
733  // Detection of pile-up from previous bunch by comparing event pedestal with reference
734  // (reference can be orbit or QC). If pile-up is detected we use orbit pedestal
735  // instead of event pedestal
736  // TODO: pedestal event could have a TM..
737  if (hasEvPed && (mSource[ich] == PedOr || mSource[ich] == PedQC)) {
738  auto pedref = mOffset[ich];
739  if (evPed > pedref && (evPed - pedref) > mRopt->ped_thr_hi[ich]) {
740  // Anomalous offset (put a warning but use event pedestal)
741  rec.offPed[ich] = true;
742  } else if (evPed < pedref && (pedref - evPed) > mRopt->ped_thr_lo[ich]) {
743  // Possible presence of pile-up (will need to use orbit pedestal)
744  rec.pilePed[ich] = true;
745  }
746  }
747  float myPed = std::numeric_limits<float>::infinity();
748  // TODO:
749  if (hasEvPed && rec.pilePed[ich] == false) {
750  myPed = evPed;
751  rec.adcPedEv[ich] = true;
752  } else if (mSource[ich] == PedOr) {
753  myPed = mOffset[ich];
754  rec.adcPedOr[ich] = true;
755  } else if (mSource[ich] == PedQC) {
756  myPed = mOffset[ich];
757  rec.adcPedQC[ich] = true;
758  } else {
759  rec.adcPedMissing[ich] = true;
760  }
761  if (myPed < std::numeric_limits<float>::infinity()) {
762  float sum = 0;
763  for (int is = mRopt->beg_int[ich]; is <= mRopt->end_int[ich]; is++) {
764  // TODO: pile-up correction
765  // TODO: manage signal positioned across boundary
766  sum += (myPed - float(mChData[ref[0]].data[is]));
767  }
768  rec.ezdc[ich] = sum * mRopt->energy_calib[ich];
769  } else {
770  LOGF(warn, "%d.%-4d CH %2d %s missing pedestal", rec.ir.orbit, rec.ir.bc, ich, ChannelNames[ich].data());
771  }
772  } else {
773  // This could arise from memory corruption or in the case when TDC bits are forced to 1
774  // due to a broken channel. For example Sum is broken and sum TDC is forced to 1
775  // Take a very small signal that triggers on one module and not the other
776  // 1) module 0 triggered by ZNATC is autotriggered
777  // 2) module 1 triggered by ZNATC is not triggered -> Sum is missing
778  // 3) sum TDC is forced to 1 and therefore the TDC is fired even if data are not present
779  // 4) you get this error flag in reconstruction
780  // This should happen only in case of hardware fault and signals near threshold
781  // This should be mitigated by having a software threshold higher than the hardware one
782  rec.adcPedMissing[ich] = true;
783  if (mVerbosity >= DbgMedium) {
784  LOGF(warn, "%d.%-4d CH %2d %s ADC missing, TDC present", rec.ir.orbit, rec.ir.bc, ich, ChannelNames[ich].data());
785  }
786  }
787  }
788  // Shift bunches by 1 taking into account bunch crossing difference
789  // TODO
790  // This is a simple shift by 1 position
791  if (ibun != iend) {
792  for (int ibcr = NBCReadOut - 1; ibcr > 0; ibcr--) {
793  ref[ibcr] = ref[ibcr - 1];
794  hasHit[ibcr] = hasHit[ibcr - 1];
795  hasAuto0[ibcr] = hasAuto0[ibcr - 1];
796  hasAutoM[ibcr] = hasAutoM[ibcr - 1];
797  hasFired[ibcr] = hasFired[ibcr - 1];
798  }
799  }
800  } // Loop on bunches
801  } // Loop on channels
802  if (mTreeDbg) {
803  for (int ibun = ibeg; ibun <= iend; ibun++) {
804  auto& rec = mReco[ibun];
805  mRec = rec;
806  mTDbg->Fill();
807  }
808  }
809  return 0;

854 {
855 #ifdef O2_ZDC_DEBUG
856  LOG(info) << __func__ << "(itdc=" << itdc << "[" << ChannelNames[TDCSignal[itdc]] << "], " << ibeg << ", " << iend << "): " << mReco[ibeg].ir.orbit << "." << mReco[ibeg].ir.bc << " - " << mReco[iend].ir.orbit << "." << mReco[iend].ir.bc;
857 #endif
858  // Extracting TDC information for TDC number itdc, in consecutive bunches from ibeg to iend
859  int nbun = iend - ibeg + 1;
860  int maxs2 = NTimeBinsPerBC * nbun - 1;
861  int shift = mRopt->tsh[itdc];
862  int thr = mRopt->tth[itdc];
863 
864  int is1 = 0, is2 = 1;
865  uint8_t isfired = 0;
866 #ifdef O2_ZDC_DEBUG
867  int16_t m[3] = {0};
868  int16_t s[3] = {0};
869 #endif
870  int it1 = 0, it2 = 0, ib1 = -1, ib2 = -1;
871  for (;;) {
872  // Shift data
873  isfired = isfired << 1;
874 #ifdef O2_ZDC_DEBUG
875  for (int i = 2; i > 0; i--) {
876  m[i] = m[i - 1];
877  s[i] = s[i - 1];
878  }
879 #endif
880  // Bunches and samples that are used in the difference
881  int b1 = ibeg + is1 / NTimeBinsPerBC;
882  int b2 = ibeg + is2 / NTimeBinsPerBC;
883  int s1 = is1 % NTimeBinsPerBC;
884  int s2 = is2 % NTimeBinsPerBC;
885  auto ref_m = mReco[b1].ref[TDCSignal[itdc]]; // reference to minuend
886  auto ref_s = mReco[b2].ref[TDCSignal[itdc]]; // reference to subtrahend
887  // Check data consistency before computing difference
888  if (ref_m == ZDCRefInitVal || ref_s == ZDCRefInitVal) {
889  LOG(fatal) << "Missing information for bunch crossing";
890  return;
891  }
892  // Check that bunch crossings are indeed the same or consecutive
893  auto bcd = mReco[b2].ir.differenceInBC(mReco[b1].ir);
894  if (bcd != 0 && bcd != 1) {
895  LOG(fatal) << __func__ << ": large bunch crossing difference " << mReco[b1].ir.orbit << "." << mReco[b1].ir.bc << " followed by " << mReco[b2].ir.orbit << "." << mReco[b2].ir.bc;
896  return;
897  }
898  int diff = mChData[ref_m].data[s1] - mChData[ref_s].data[s2];
899  // Triple trigger condition
900 #ifdef O2_ZDC_DEBUG
901  m[0] = mChData[ref_m].data[s1];
902  s[0] = mChData[ref_s].data[s2];
903 #endif
904  if (diff > thr) {
905  isfired = isfired | 0x1;
906  if ((isfired & mTriggerCondition) == mTriggerCondition) {
907  // Fired bit is assigned to the second sample, i.e. to the one that can identify the
908  // signal peak position
909  mReco[b2].fired[itdc] |= mMask[s2];
910 #ifdef O2_ZDC_DEBUG
911  if (mTriggerCondition == 0x7) {
912  printf("0x7 TDC %d[%s] Fired @ %u.%u.s%02u (%5d-%5d)=%5d>%2d && (%5d-%5d)=%5d>%2d && (s%02d:%-5d-s%02d:%-5d)=%5d>%2d\n",
913  itdc, ChannelNames[TDCSignal[itdc]].data(), mReco[b2].ir.orbit, mReco[b2].ir.bc, s2,
914  m[2], s[2], (m[2] - s[2]), thr,
915  m[1], s[1], (m[1] - s[1]), thr,
916  s1, m[0], s2, s[0], diff, thr);
917  } else if (mTriggerCondition == 0x3) {
918  printf("0x3 TDC %d[%s] Fired @ %u.%u.s%02u (%5d-%5d)=%5d>%2d && (s%02d:%-5d-s%02d:(%-5d))=%5d>%2d\n",
919  itdc, ChannelNames[TDCSignal[itdc]].data(), mReco[b2].ir.orbit, mReco[b2].ir.bc, s2,
920  m[1], s[1], (m[1] - s[1]), thr,
921  s1, m[0], s2, s[0], diff, thr);
922  } else if (mTriggerCondition == 0x1) {
923  printf("0x1 TDC %d[%s] Fired @ %u.%u.s%02u (%d-(%d))=%d>%d && (%d-(%d))=%d>%d && (s%d:%d-s%d:(%d))=%d>%d\n",
924  itdc, ChannelNames[TDCSignal[itdc]].data(), mReco[b2].ir.orbit, mReco[b2].ir.bc, s2,
925  s1, m[0], s2, s[0], diff, thr);
926  }
927 #endif
928  }
929  }
930  if (is2 >= shift) {
931  is1++;
932  }
933  if (is2 < maxs2) {
934  is2++;
935  }
936  if (is1 == maxs2) {
937  break;
938  }
939  }
940  interpolate(itdc, ibeg, iend);

1209  // can occur in these two regions

1323  // performed after all signals have been identified
1324  if (mSource[isig] != PedND) {
1325  amp = mOffset[isig] - amp;
1326  } else {

1336  }

1339 #ifndef O2_ZDC_INTERP_DEBUG

1341  // setPoint(itdc, ibeg, iend, isam);

1343 #else

1349  isam_amp = isam;
1350  }
1351  }
1352  }

1356  if (amp <= ADCMax) {

1554 void DigiReco::correctTDCPile()
1555 {
1556  // Pile-up correction for TDCs
1557  // FEE acquires data in two modes: triggered and continuous
1558  // In triggered mode minimum two and up to four bunches are transferred for each ALICE trigger
1559  // In continuous mode two bunches are transferred for each autotrigger
1560  // Reconstruction is performed for consecutive bunches
1561  // There is no issue with gaps for triggered mode because the trigger condition
1562  // A0 || A1 || (A2 && (T0 || TM)) || (A3 && T0) one can have the following sequences
1563  // where: T is autotrigger, A is ALICE trigger, P is a previous bunch, - is skipped
1564  // ---PA A0 || A1
1565  // ---TA A0 || A1
1566  // --TPA A2 && T0
1567  // -TPPA A2 && TM A3 && T0
1568  // On the other hand, in autotrigger mode one can have a gap
1569  // ---PT
1570  // --PTT
1571  // -PTPT
1572  // PT-PT
1573  // therefore we have to look for an interaction outside the range of consecutive bunch
1574  // crossings that is used in reconstruction. Therefore the correction is done outside
1575  // reconstruction loop
1576  // In case TDC correction parameters are missing (e.g. mTDCCorr==0) then
1577  // pile-up is flagged but not corrected for
