31 void DigiRecoTest::init()
32 {
33  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
34  if (mCCDBServer.size() > 0) {
35  mgr.setURL(mCCDBServer);
36  mDigi.setCCDBServer(mCCDBServer);
37  }
38  mDigi.init();
39 
40  LOG(info) << "Initialization of ZDC Test reconstruction " << mgr.getURL();
41  // CCDB server has been initialized in DigitizerTest
42  long timeStamp = mgr.getTimestamp();
43 
44  auto* moduleConfig =
45  mgr.get<o2::zdc::ModuleConfig>(o2::zdc::CCDBPathConfigModule);
46  if (!moduleConfig) {
47  LOG(fatal) << "Missing ModuleConfig configuration object @ " << o2::zdc::CCDBPathConfigModule;
48  return;
49  }
50  LOG(info) << "Loaded ModuleConfig for timestamp " << timeStamp;
51  moduleConfig->print();
52 
53  // Configuration parameters for ZDC reconstruction
54  auto* recoConfigZDC =
55  mgr.get<o2::zdc::RecoConfigZDC>(o2::zdc::CCDBPathRecoConfigZDC);
56  if (!recoConfigZDC) {
57  LOG(fatal) << "Missing RecoConfigZDC object";
58  return;
59  }
60  LOG(info) << "Loaded RecoConfigZDC for timestamp " << timeStamp;
61  recoConfigZDC->print();
62 
63  // TDC centering
64  auto* tdcParam = mgr.get<o2::zdc::ZDCTDCParam>(o2::zdc::CCDBPathTDCCalib);
65  if (!tdcParam) {
66  LOG(fatal) << "Missing ZDCTDCParam calibration object";
67  return;
68  }
69  LOG(info) << "Loaded TDC centering ZDCTDCParam for timestamp " << timeStamp;
70  tdcParam->print();
71 
72  // TDC correction parameters
73  auto* tdcCorr = mgr.get<o2::zdc::ZDCTDCCorr>(o2::zdc::CCDBPathTDCCorr);
74  if (!tdcCorr) {
75  LOG(fatal) << "Missing ZDCTDCCorr calibration object - no correction is applied";
76  } else {
77  if (mVerbosity > DbgZero) {
78  LOG(info) << "Loaded TDC correction parameters for timestamp " << timeStamp;
79  tdcCorr->print();
80  }
81  }
82 
83  // Energy calibration
84  auto* energyParam =
85  mgr.get<o2::zdc::ZDCEnergyParam>(o2::zdc::CCDBPathEnergyCalib);
86  if (!energyParam) {
87  LOG(warning)
88  << "Missing ZDCEnergyParam calibration object - using default";
89  } else {
90  LOG(info) << "Loaded Energy calibration ZDCEnergyParam for timestamp "
91  << timeStamp;
92  energyParam->print();
93  }
94 
95  // Tower calibration
96  auto* towerParam =
97  mgr.get<o2::zdc::ZDCTowerParam>(o2::zdc::CCDBPathTowerCalib);
98  if (!towerParam) {
99  LOG(warning)
100  << "Missing ZDCTowerParam calibration object - using default";
101  } else {
102  LOG(info) << "Loaded Tower calibration ZDCTowerParam for timestamp "
103  << timeStamp;
104  towerParam->print();
105  }
106 
107  mDR.setModuleConfig(moduleConfig);
108  mDR.setRecoConfigZDC(recoConfigZDC);
109  mDR.setTDCParam(tdcParam);
110  mDR.setTDCCorr(tdcCorr);
111  mDR.setEnergyParam(energyParam);
112  mDR.setTowerParam(towerParam);
113  mDR.setVerbosity(mVerbosity);
114  mDR.init();
115 
116  const uint32_t* mask = mDR.getChMask();
117  for (uint32_t ic = 0; ic < NChannels; ic++) {
118  mDigi.setMask(ic, mask[ic]);
119  }
120 } // init
