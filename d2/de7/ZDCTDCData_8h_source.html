47  {
48  id = ida & 0x0f;
49  id = id | (isbeg ? 0x80 : 0x00);
50  id = id | (isend ? 0x40 : 0x00);
51 
52  auto TDCVal = std::nearbyint(vala);
53  auto TDCAmp = std::nearbyint(ampa);
54 
55  if (TDCVal < kMinShort) {
56  LOG(error) << __func__ << " TDCVal " << int(ida) << " " << ChannelNames[ida] << " = " << TDCVal << " is out of range";

58  }
59  if (TDCVal > kMaxShort) {
60  LOG(error) << __func__ << " TDCVal " << int(ida) << " " << ChannelNames[ida] << " = " << TDCVal << " is out of range";
61  TDCVal = kMaxShort;
62  }

64  LOG(error) << __func__ << " TDCAmp " << int(ida) << " " << ChannelNames[ida] << " = " << TDCAmp << " is out of range";
65  TDCAmp = kMinShort;
66  }

68  LOG(error) << __func__ << " TDCAmp " << int(ida) << " " << ChannelNames[ida] << " = " << TDCAmp << " is out of range";

70  }
71 
72  val = TDCVal;
