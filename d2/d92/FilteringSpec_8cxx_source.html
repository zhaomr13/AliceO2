93 
94  auto primVer2TRefs = recoData.getPrimaryVertexMatchedTrackRefs();

101  auto fddChData = recoData.getFDDChannelsData();

105  auto fv0ChData = recoData.getFV0ChannelsData();
106  auto fv0RecPoints = recoData.getFV0RecPoints();
107 
108  auto zdcEnergies = recoData.getZDCEnergy();
109  auto zdcBCRecData = recoData.getZDCBCRecData();
110  auto zdcTDCData = recoData.getZDCTDCData();
111 
112  // get calo information
113  auto caloEMCCells = recoData.getEMCALCells();
114  auto caloEMCCellsTRGR = recoData.getEMCALTriggers();
115 
116  auto ctpDigits = recoData.getCTPDigits();
117  // std::unique_ptr<o2::steer::MCKinematicsReader> mcReader = std::make_unique<o2::steer::MCKinematicsReader>("collisioncontext.root");
118  */
119 
120  // process tracks associated with vertices, note that the last entry corresponds to orphan tracks (no vertex)
121  for (const auto& trackRef : primVer2TRefs) {
122  processTracksOfVertex(trackRef, recoData);
123  }
124 
125  if (mNeedToSave) {
126  fillData(recoData);
127  auto dref = pc.inputs().getFirstValid(true);
128  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(dref);

134

136  clear(); // clear caches, safe after the snapshot
137  }
138

162 void FilteringSpec::fillData(const o2::globaltracking::RecoContainer& recoData)
163 {
164  // actual data filling
165 
166  // ITS tracks
167  if (!mITSTrackIDCache.empty()) {
168  const auto tracksOrig = recoData.getITSTracks();
169  const auto tracksOrigLbl = recoData.getITSTracksMCLabels();
170  const auto clusRefOrig = recoData.getITSTracksClusterRefs();
171  const auto rofsOrig = recoData.getITSTracksROFRecords();
172  auto trIt = mITSTrackIDCache.begin(); // tracks sorted in their ID
173  for (unsigned irof = 0; irof < rofsOrig.size() && trIt != mITSTrackIDCache.end(); irof++) {
174  const auto& rofOrig = rofsOrig[irof];
175  int startID = rofOrig.getFirstEntry(), endID = startID + rofOrig.getNEntries();
176  if (trIt->first >= endID) {
177  continue; // nothing from this ROF is selected
178  }
179  auto& rofSave = mFTF.ITSTrackROFs.emplace_back(rofOrig);
180  rofSave.setFirstEntry(mFTF.ITSTracks.size());
181  while (trIt != mITSTrackIDCache.end() && trIt->first < endID) {
182  trIt->second = mFTF.ITSTracks.size(); // this for the further bookkeeping?
183  const auto& trOr = tracksOrig[trIt->first];
184  auto& trSave = mFTF.ITSTracks.emplace_back(trOr);
185  trSave.setFirstClusterEntry(mFTF.ITSClusterIndices.size()); // N cluster entries is set correctly at creation
186  for (int i = 0; i < trOr.getNClusters(); i++) {
187  int clID = trOr.getClusterEntry(i);
188  mFTF.ITSClusterIndices.push_back(clID); // later will be remapped to actually stored cluster indices
189  mITSClusterIDCache[clID] = 0; // flag used cluster
190  }
191  if (mUseMC) {
192  mFTF.ITSTrackMCTruth.push_back(tracksOrigLbl[trIt->first]);
193  }
194  trIt++;
195  }
196  rofSave.setNEntries(mFTF.ITSTracks.size() - rofSave.getFirstEntry());
197  }
198  }
199  // ITS clusters info for selected tracks
200  if (!mITSClusterIDCache.empty()) {
201  const auto clusOrig = recoData.getITSClusters();
202  const auto pattOrig = recoData.getITSClustersPatterns();
203  const auto rofsOrig = recoData.getITSClustersROFRecords();
204  auto clIt = mITSClusterIDCache.begin(); // clusters sorted in their ID
205  auto pattItOrig = pattOrig.begin(), pattItOrigPrev = pattItOrig;
206  for (unsigned irof = 0; irof < rofsOrig.size() && clIt != mITSClusterIDCache.end(); irof++) {
207  const auto& rofOrig = rofsOrig[irof];
208  int startID = rofOrig.getFirstEntry(), endID = startID + rofOrig.getNEntries();
209  if (clIt->first >= endID) {
210  continue; // nothing from this ROF is selected
211  }
212  auto& rofSave = mFTF.ITSClusterROFs.emplace_back(rofOrig);
213  rofSave.setFirstEntry(mFTF.ITSTracks.size());
214  while (clIt != mITSClusterIDCache.end() && clIt->first < endID) {
215  clIt->second = mFTF.ITSClusters.size(); // new index of the stored cluster
216  const auto& clOr = clusOrig[clIt->first];
217  auto& clSave = mFTF.ITSClusters.emplace_back(clOr);
218  // cluster pattern
219  auto pattID = clOr.getPatternID();
220  o2::itsmft::ClusterPattern patt;
221  if (pattID == o2::itsmft::CompCluster::InvalidPatternID || mDictITS->isGroup(pattID)) {
222  patt.acquirePattern(pattItOrig);
223  }
224  while (pattItOrigPrev != pattItOrig) { // the difference, if any, is the explicitly stored pattern
225  mFTF.ITSClusterPatterns.push_back(*pattItOrigPrev);
226  pattItOrigPrev++;
227  }
228  clIt++;
229  }
230  rofSave.setNEntries(mFTF.ITSClusters.size() - rofSave.getFirstEntry());
231  }
232  }
233  // now remap cluster indices to stored values
234  for (auto& clID : mFTF.ITSClusterIndices) {
235  clID = mITSClusterIDCache[clID];
236  }
237 }

239 void FilteringSpec::processTracksOfVertex(const o2::dataformats::VtxTrackRef& trackRef, const o2::globaltracking::RecoContainer& recoData)
240 {
241  auto GIndices = recoData.getPrimaryVertexMatchedTracks(); // Global IDs of all tracks
242  int vtxID = trackRef.getVtxID(); // -1 means that we are processing orphan tracks
243 
244  for (int src = GIndex::NSources; src--;) { // loop over all possible types of tracks
245  int start = trackRef.getFirstEntryOfSource(src);
246  int end = start + trackRef.getEntriesOfSource(src);
247  for (int ti = start; ti < end; ti++) {
248  auto& trackIndex = GIndices[ti];
249  if (GIndex::includesSource(src, mInputSources)) { // should we consider this source
250 
251  if (trackIndex.isAmbiguous()) { // this trak was matched to multiple vertices
252  const auto res = mGIDToTableID.find(trackIndex);
253  if (res != mGIDToTableID.end()) { // and was already processed
254  if (res->second < 0) { // was selected, just register its vertex
255  // registerVertex // FIXME: will be done once processing of all track types is complete
256  }
257  continue;
258  }
259  }
260  // here we select barrel tracks only (they must have TPC or ITS contributions)
261  if (!trackIndex.includesDet(DetID::ITS) && !trackIndex.includesDet(DetID::TPC)) {
262  continue;
263  }
264 
265  int selRes = processBarrelTrack(trackIndex, recoData); // was track selected?
266  if (trackIndex.isAmbiguous()) { // remember decision on this track, if will appear again
267  mGIDToTableID[trackIndex] = selRes; // negative answer means rejection
268  }
269  }
270  }
271  }

274 int FilteringSpec::processBarrelTrack(GIndex idx, const o2::globaltracking::RecoContainer& recoData)
275 {
276  int res = selectTrack(idx, recoData);
277  if (res < 0) {
278  return res;
279  }
280  mNeedToSave = true;
281  auto contributorsGID = recoData.getSingleDetectorRefs(idx);
282  //
283  // here is example of processing the ITS track
284  if (contributorsGID[GID::ITS].isIndexSet()) {
285  mITSTrackIDCache[contributorsGID[GID::ITS]] = 0;
286  }
287 
288  return res;
289 }

291 bool FilteringSpec::selectTrack(GIndex id, const o2::globaltracking::RecoContainer& recoData)
292 {
293  o2::track::TrackParCov t;
294  auto src = id.getSource();
295  if (src == GID::ITSTPCTRDTOF) {
296  t = recoData.getTrack<o2::track::TrackParCov>(recoData.getITSTPCTRDTOFMatches()[id].getTrackRef()); // ITSTPCTRDTOF is ITSTPCTRD + TOF cluster
297  } else if (src == GID::TPCTRDTOF) {
298  t = recoData.getTrack<o2::track::TrackParCov>(recoData.getTPCTRDTOFMatches()[id].getTrackRef()); // TPCTRDTOF is TPCTRD + TOF cluster
299  } else if (src == GID::ITSTPCTOF) {
300  t = recoData.getTrack<o2::track::TrackParCov>(recoData.getTOFMatch(id).getTrackRef()); // ITSTPCTOF is ITSTPC + TOF cluster
301  } else { // for the rest, get the track directly
302  t = recoData.getTrack<o2::track::TrackParCov>(id);
303  }
304  // select on track kinematics, for example, on pT
305  if (t.getPt() < 2.) {
306  return false;
307  }
308  return true;
309 }

321 DataProcessorSpec getDataFilteringSpec(GID::mask_t src, bool enableSV, bool useMC)
322 {
323  std::vector<OutputSpec> outputs;
324  outputs.emplace_back("GLO", "FILTERED_RECO_TF", 0, o2::framework::Lifetime::Sporadic); // sporadic to avoid sensing TFs which had no selected stuff
325 
326  auto dataRequest = std::make_shared<DataRequest>();
327 
328  dataRequest->requestTracks(src, useMC);
329  dataRequest->requestPrimaryVertertices(useMC);
330  if (src[GID::CTP]) {
331  LOGF(info, "Requesting CTP digits");
332  dataRequest->requestCTPDigits(useMC);
333  }
334  if (enableSV) {
335  dataRequest->requestSecondaryVertertices(useMC);
336  }
337  if (src[GID::TPC]) {
338  dataRequest->requestClusters(GIndex::getSourcesMask("TPC"), false); // no need to ask for TOF clusters as they are requested with TOF tracks
339  }
340  if (src[GID::EMC]) {
341  dataRequest->requestEMCALCells(useMC);
342  }
343 
344  dataRequest->inputs.emplace_back("itsDict", "ITS", "CLUSTERDICT", 0, Lifetime::Condition, ccdbParamSpec("ITS/Calib/ClusterDictionary"));
345 
346  return DataProcessorSpec{
347  "reco-data-filter",
348  dataRequest->inputs,
349  outputs,
350  AlgorithmSpec{adaptFromTask<FilteringSpec>(src, dataRequest, enableSV, useMC)},
351  Options{/*ConfigParamSpec{"reco-mctracks-only", VariantType::Int, 0, {"Store only reconstructed MC tracks and their mothers/daughters. 0 -- off, != 0 -- on"}}*/}};
352 }
