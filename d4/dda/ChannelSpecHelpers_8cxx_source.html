47 bool isIPAddress(const std::string& address)
48 {
49  std::regex ipv4_regex("^(([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])\\.){3}([0-9]|[1-9][0-9]|1[0-9]{2}|2[0-4][0-9]|25[0-5])$");
50  if (std::regex_match(address, ipv4_regex)) {
51  return true;
52  }
53  return false;

56 void OutputChannelSpecConfigParser::property(std::string_view key, std::string_view value)
57 {
58  std::string valueStr = value.data();
59  if (key == "address") {
60  auto parseAddress = [v = std::string(value), &outputChannelSpec = specs.back()]() {
61  auto value = v;
62  std::string protocol = "tcp";
63  std::string hostname = "127.0.0.1";
64  std::string port = "9090";
65  auto pos = value.find("://");
66  if (pos != std::string::npos) {
67  protocol = value.substr(0, pos);
68  value = value.substr(pos + 3);
69  }
70  if (protocol == "tcp") {
71  pos = value.find(':');
72  if (pos != std::string::npos) {
73  hostname = value.substr(0, pos);
74  value = value.substr(pos + 1);
75  } else {
76  throw runtime_error_f("Port not found in address '%s'", v.c_str());
77  }
78  port = value;
79  if (isIPAddress(hostname) == false) {
80  throw runtime_error_f("Invalid ip address '%s'", hostname.c_str());
81  }
82  outputChannelSpec.hostname = hostname;
83  outputChannelSpec.port = std::stoi(port);
84  outputChannelSpec.protocol = ChannelProtocol::Network;
85  } else if (protocol == "ipc") {
86  outputChannelSpec.hostname = value;
87  outputChannelSpec.port = 0;
88  outputChannelSpec.protocol = ChannelProtocol::IPC;
89  } else {
90  throw runtime_error_f("Unknown protocol '%s'", protocol.c_str());
91  }
92  };
93  parseAddress();
94  }
95  auto& outputChannelSpec = specs.back();
96  if (key == "name") {
97  outputChannelSpec.name = value;
98  } else if (key == "type" && value == "pub") {
99  outputChannelSpec.type = ChannelType::Pub;
100  } else if (key == "type" && value == "sub") {
101  outputChannelSpec.type = ChannelType::Sub;
102  } else if (key == "type" && value == "push") {
103  outputChannelSpec.type = ChannelType::Push;
104  } else if (key == "type" && value == "pull") {
105  outputChannelSpec.type = ChannelType::Pull;
106  } else if (key == "type" && value == "pair") {
107  outputChannelSpec.type = ChannelType::Pair;
108  } else if (key == "method" && value == "bind") {
109  outputChannelSpec.method = ChannelMethod::Bind;
110  } else if (key == "method" && value == "connect") {
111  outputChannelSpec.method = ChannelMethod::Connect;
112  } else if (key == "rateLogging") {
113  outputChannelSpec.rateLogging = std::stoi(valueStr);
114  } else if (key == "recvBufSize") {
115  outputChannelSpec.recvBufferSize = std::stoi(valueStr);
116  } else if (key == "sendBufSize") {
117  outputChannelSpec.recvBufferSize = std::stoi(valueStr);
118  }
119 }

156 std::string composeIPCName(std::string const& prefix, std::string const& hostname, short port)
157 {
158  if (prefix.empty() == false && prefix[0] == '@') {
159  return fmt::format("ipc://{}{}_{},transport=shmem", prefix, hostname, port);
160  }
161  if (prefix.empty() == false && prefix.back() == '/') {
162  return fmt::format("ipc://{}{}_{},transport=shmem", prefix, hostname, port);
163  }
164  return fmt::format("ipc://{}/{}_{},transport=shmem", prefix, hostname, port);
165 }

215 void ChannelSpecHelpers::parseChannelConfig(char const* config, FairMQChannelConfigParser& handler)
216 {
217  ChannelConfigParserState state = ChannelConfigParserState::BEGIN;
218  char const* cur = config;
219  char const* next = config;
220  std::string_view key;
221  std::string_view value;
222  char const* nameKey = "name";
223  char const* lastError = "bad configuation string";
224 
225  while (true) {
226  switch (state) {
227  case ChannelConfigParserState::BEGIN: {
228  if (*cur == '\0') {
229  lastError = "empty config string";
230  state = ChannelConfigParserState::ERROR;
231  } else if (!isalpha(*cur)) {
232  lastError = "first character is not alphabetic";
233  state = ChannelConfigParserState::ERROR;
234  } else {
235  state = ChannelConfigParserState::BEGIN_CHANNEL;
236  }
237  break;
238  }
239  case ChannelConfigParserState::BEGIN_CHANNEL: {
240  next = strpbrk(cur, ":=;,");
241  if (*next == ';' || *next == ',') {
242  lastError = "expected channel name";
243  state = ChannelConfigParserState::ERROR;
244  break;
245  } else if (*next == ':') {
246  handler.beginChannel();
247  key = std::string_view(nameKey, 4);
248  value = std::string_view(cur, next - cur);
249  handler.property(key, value);
250  state = ChannelConfigParserState::BEGIN_KEY;
251  cur = next + 1;
252  if (*cur == '\0') {
253  state = ChannelConfigParserState::END_CHANNEL;
254  } else {
255  state = ChannelConfigParserState::BEGIN_KEY;
256  }
257  break;
258  }
259  handler.beginChannel();
260  state = ChannelConfigParserState::BEGIN_KEY;
261  break;
262  }
263  case ChannelConfigParserState::BEGIN_KEY: {
264  next = strchr(cur, '=');
265  if (next == nullptr) {
266  lastError = "expected '='";
267  state = ChannelConfigParserState::ERROR;
268  } else {
269  key = std::string_view(cur, next - cur);
270  state = ChannelConfigParserState::BEGIN_VALUE;
271  cur = next + 1;
272  }
273  break;
274  }
275  case ChannelConfigParserState::BEGIN_VALUE: {
276  next = strpbrk(cur, ";,");
277  if (next == nullptr) {
278  size_t l = strlen(cur);
279  value = std::string_view(cur, l);
280  state = ChannelConfigParserState::END_VALUE;
281  cur = cur + l;
282  } else if (*next == ';') {
283  value = std::string_view(cur, next - cur);
284  state = ChannelConfigParserState::END_CHANNEL;
285  cur = next;
286  } else if (*next == ',') {
287  value = std::string_view(cur, next - cur);
288  state = ChannelConfigParserState::END_VALUE;
289  cur = next;
290  }
291  handler.property(key, value);
292  break;
293  }
294  case ChannelConfigParserState::END_VALUE: {
295  if (*cur == '\0') {
296  state = ChannelConfigParserState::END_CHANNEL;
297  } else if (*cur == ',') {
298  state = ChannelConfigParserState::BEGIN_KEY;
299  cur++;
300  } else if (*cur == ';') {
301  state = ChannelConfigParserState::END_CHANNEL;
302  cur++;
303  }
304  break;
305  }
306  case ChannelConfigParserState::END_CHANNEL: {
307  handler.endChannel();
308  if (*cur == '\0') {
309  state = ChannelConfigParserState::END;
310  } else if (*cur == ';') {
311  state = ChannelConfigParserState::BEGIN_CHANNEL;
312  cur++;
313  } else {
314  lastError = "expected ';'";
315  state = ChannelConfigParserState::ERROR;
316  }
317  break;
318  }
319  case ChannelConfigParserState::END: {
320  return;
321  }
322  case ChannelConfigParserState::ERROR: {
323  throw runtime_error_f("Unable to parse channel config: %s", lastError);
324  }
325  }
326  }
327 }
