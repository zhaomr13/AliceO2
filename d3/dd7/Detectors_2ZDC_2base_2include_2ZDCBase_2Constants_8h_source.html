331 //< map channelID to detector/tower
332 constexpr int toDet(int channel, int& tower)
333 {
334  if (channel < toChannel(ZNC, 0)) {
335  tower = channel % NChannelsZN;
336  return DetIDOffs + channel / NChannelsZP;
337  } else {
338  channel -= toChannel(ZNC, 0);
339  tower = channel % NChannelsZN;
340  return ZNC + channel / NChannelsZP;
341  }
342 }
