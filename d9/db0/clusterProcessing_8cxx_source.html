145 // Remove hits/seeds according to w-magnitude
146 // and hits/seeds proximity
147 // Return the new # of components (hits/seeds)
148 int filterEMModel(double* theta, int K, Mask_t* maskFilteredTheta)
149 {
150  // w cut-off
151  double cutOff = 1.0 / K * wCutoff;
152  //
153  double* w = getW(theta, K);
154  double* muX = getMuX(theta, K);
155  double* muY = getMuY(theta, K);
156  double* varX = getVarX(theta, K);
157  double* varY = getVarY(theta, K);
158  //
159  // Sort w (index sorting)
160  // Indexes for sorting
161  int index[K];
162  for (int k = 0; k < K; k++) {
163  index[k] = k;
164  }
165  std::sort(index, &index[K], [=](int a, int b) { return w[a] > w[b]; });
166  // Test ??? to supress
167  if (VERBOSE) {
168  vectorPrint(" sort w", w, K);
169  vectorPrintInt(" sort w-indexes", index, K);
170  }
171  // Mode constant variance for all hits/seeds
172  double sigX = sqrt(varX[0]);
173  double sigY = sqrt(varY[0]);
174  //
175  // Reorder theta in fct of the w sort
176  // and suppress closed theta[k] (hit/seeds)
177  // ( w is set to zero)
178  for (int i = 0; i < K; i++) {
179  double muXi = muX[index[i]];
180  double muYi = muY[index[i]];
181  // Suppress low w
182  if (w[index[i]] > cutOff) {
183  for (int j = i + 1; j < K; j++) {
184  double dMuX = fabs(muXi - muX[index[j]]);
185  double dMuY = fabs(muYi - muY[index[j]]);
186  // Suppress closed hits/seeds
187  if ((dMuX < sigX) && (dMuY < sigY)) {
188  w[index[j]] = 0;
189  }
190  }
191  } else {
192  w[index[i]] = 0;
193  }
194  }
195 
196  // Build the mask w[k] >= 0.0
197  for (int k = 0; k < K; k++) {
198  maskFilteredTheta[k] = (w[k] > 0.0);
199  }
200  int newK = vectorSumShort(maskFilteredTheta, K);
201  return newK;
202 }

204 int filterFitModelOnClusterRegion(const double* xyDxy, int N, double* theta, int K, Mask_t* maskFilteredTheta)
205 {
206  // Spatial filter
207  //
208  const double* x = getConstX(xyDxy, N);
209  const double* y = getConstY(xyDxy, N);
210  const double* dx = getConstDX(xyDxy, N);
211  const double* dy = getConstDY(xyDxy, N);
212  // Min Max pads
213  double xyTmp[N];
214  int kSpacialFilter = 0;
215  vectorAddVector(x, -1.0, dx, N, xyTmp);
216  double xMin = vectorMin(xyTmp, N);
217  vectorAddVector(x, +1.0, dx, N, xyTmp);
218  double xMax = vectorMax(xyTmp, N);
219  vectorAddVector(y, -1.0, dy, N, xyTmp);
220  double yMin = vectorMin(xyTmp, N);
221  vectorAddVector(y, +1.0, dy, N, xyTmp);
222  double yMax = vectorMax(xyTmp, N);
223  double* muX = getMuX(theta, K);
224  double* muY = getMuY(theta, K);
225  for (int k = 0; k < K; k++) {
226  maskFilteredTheta[k] = 0;
227  if ((muX[k] > xMin) && (muX[k] < xMax)) {
228  if ((muY[k] > yMin) && (muY[k] < yMax)) {
229  maskFilteredTheta[k] = 1;
230  kSpacialFilter++;
231  }
232  }
233  }
234  if ((VERBOSE > 0) && (kSpacialFilter != K)) {
235  printf("---> Spacial Filter; removing %d hit\n", K - kSpacialFilter);
236  }
237  // W filter
238  // w cut-off
239  double cutOff = 0.02 / kSpacialFilter;
240  //
241  double* w_ = getW(theta, K);
242  double w[K];
243  double wSum = 0.0;
244  // Normalize new w
245  for (int k = 0; k < K; k++) {
246  wSum += (maskFilteredTheta[k] * w_[k]);
247  }
248  int kWFilter = 0;
249  double norm = 1.0 / wSum;
250  for (int k = 0; k < K; k++) {
251  w[k] = maskFilteredTheta[k] * w_[k] * norm;
252  maskFilteredTheta[k] = maskFilteredTheta[k] && (w[k] > cutOff);
253  kWFilter += (maskFilteredTheta[k] && (w[k] > cutOff));
254  }
255  if ((VERBOSE > 0) && (kSpacialFilter > kWFilter)) {
256  printf("---> At least one hit, w[k] < (0.05 / K) = %8.4f); removing %d hit\n", cutOff, kSpacialFilter - kWFilter);
257  }
258  return kWFilter;
259 }

261 int filterFitModelOnSpaceVariations(const double* theta0, int K0, double* theta, int K, Mask_t* maskFilteredTheta)
262 {
263  // K is the same for theta0 & theta
264  // Spatial filter
265  //
266  int kSpacialFilter = 0;
267  const double* mu0X = getConstMuX(theta0, K0);

273  double xTmp[K0], yTmp[K0];
274  for (int k = 0; k < K; k++) {
275  maskFilteredTheta[k] = 0;
276  // Find the neighbour
277  vectorAddScalar(mu0X, -muX[k], K0, xTmp);
278  vectorMultVector(xTmp, xTmp, K, xTmp);
279  vectorAddScalar(mu0Y, -muY[k], K0, yTmp);
280  vectorMultVector(yTmp, yTmp, K, yTmp);

284  double xMin = mu0X[kMin] - 4 * mu0Dx[kMin];
285  double xMax = mu0X[kMin] + 4 * mu0Dx[kMin];
286  double yMin = mu0Y[kMin] - 4 * mu0Dy[kMin];
287  double yMax = mu0Y[kMin] + 4 * mu0Dy[kMin];
288  if (((muX[k] > xMin) && (muX[k] < xMax)) && ((muY[k] > yMin) && (muY[k] < yMax))) {
