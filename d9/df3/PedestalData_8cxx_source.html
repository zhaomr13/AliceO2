41 void PedestalData::fill(gsl::span<const PedestalDigit> digits)
42 {
43  bool mDebug = false;
44 
45  for (auto& d : digits) {
46  uint16_t solarId = d.getSolarId();
47  uint8_t dsId = d.getDsId();
48  uint8_t channel = d.getChannel();
49 
50  auto iPedestal = mPedestals.find(solarId);
51 
52  if (iPedestal == mPedestals.end()) {
53  auto iPedestalsNew = mPedestals.emplace(std::make_pair(solarId, initPedestalMatrix(solarId)));
54  iPedestal = iPedestalsNew.first;
55  }
56 
57  if (iPedestal == mPedestals.end()) {
58  std::cout << "[PedestalData::process] failed to insert new element\n";
59  break;
60  }
61 
62  auto& ped = iPedestal->second[dsId][channel];
63 
64  ped.dsChannelId = DsChannelId{solarId, dsId, channel};
65 
66  for (uint16_t i = 0; i < d.nofSamples(); i++) {
67  auto s = d.getSample(i);
68 
69  ped.mEntries += 1;
70  uint64_t N = ped.mEntries;
71 
72  double p0 = ped.mPedestal;
73  double p = p0 + (s - p0) / N;
74  ped.mPedestal = p;
75 
76  double M0 = ped.mVariance;
77  double M = M0 + (s - p0) * (s - p);
78  ped.mVariance = M;
79  }
80 
81  if (mDebug) {
82  std::cout << "solarId " << (int)solarId << " dsId " << (int)dsId << " ch " << (int)channel << " nsamples " << d.nofSamples()
83  << " entries " << ped.mEntries << " ped " << ped.mPedestal << " mVariance " << ped.mVariance << std::endl;
84  }
85  }
86 }
