28 std::array<int, 5> o2::zdc::fastsim::calculateChannels(Ort::Value& value)
29 {
30  std::array<float, 5> channels = {0}; // 5 photon channels
31  auto flattedImageVector = value.GetTensorMutableData<float>();
32 
33  // Model output needs to be converted with exp(x)-1 function to be valid.
34  for (int i = 0; i < 44; i++) {
35  for (int j = 0; j < 44; j++) {
36  if (i % 2 == j % 2) {
37  if (i < 22 && j < 22) {
38  channels[0] = channels[0] + (std::exp(flattedImageVector[i + j * 44]) - 1);
39  } else if (i >= 22 && j < 22) {
40  channels[1] = channels[1] + (std::exp(flattedImageVector[i + j * 44]) - 1);
41  } else if (i < 22 && j >= 22) {
42  channels[2] = channels[2] + (std::exp(flattedImageVector[i + j * 44]) - 1);
43  } else if (i >= 22 && j >= 22) {
44  channels[3] = channels[3] + (std::exp(flattedImageVector[i + j * 44]) - 1);
45  }
46  } else {
47  channels[4] = channels[4] + (std::exp(flattedImageVector[i + j * 44]) - 1);
48  }
49  }
50  }
51  std::array<int, 5> channels_integers = {0};
52  for (int ch = 0; ch < 5; ch++) {
53  channels_integers[ch] = std::round(channels[ch]);
54  }
55  return channels_integers;
56 }

66 void NeuralFastSimulation::setInputOutputData()
67 {
68  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
69  mInputNames.push_back(mSession.GetInputName(i, mAllocator));
70  }
71  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
72  mInputShapes.emplace_back(mSession.GetInputTypeInfo(i).GetTensorTypeAndShapeInfo().GetShape());
73  }
74  for (size_t i = 0; i < mSession.GetOutputCount(); ++i) {
75  mOutputNames.push_back(mSession.GetOutputName(i, mAllocator));
76  }
77 
78  // Prevent negative values from being passed as tensor shape
79  for (auto& shape : mInputShapes) {
80  for (auto& elem : shape) {
81  elem = std::abs(elem);
82  }
83  }
84 }

127 std::array<float, 9> VAEModelSimulation::scaleConditionalInput(const std::array<float, 9>& rawConditionalInput)
128 {
129  std::array<float, 9> scaledConditionalInput = {0, 0, 0, 0, 0, 0, 0, 0, 0};
130  for (int i = 0; i < 9; ++i) {

156  std::copy_n(std::make_move_iterator(means.begin()), 9, meansArray.begin());

158  return std::make_pair(meansArray, scalesArray);
