141 void ClusterFinderGEM::dumpPreCluster(ClusterDump* dumpFile, gsl::span<const Digit> digits, uint16_t bunchCrossing, uint32_t orbit, uint32_t iPreCluster)
142 {
144  // GG mPrecluster defined in other MCHPreClustering/PreClusterFinder.h
145  mSegmentation = &mapping::segmentation(digits[0].getDetID());
146 
147  // GG
148  // Allocation
149  nPads = digits.size();
150  double xyDxy[nPads * 4];
151  Mask_t saturated[nPads];
152  Mask_t cathode[nPads];
153  double padCharge[nPads];
154  // use to store in the dump file
155  uint32_t padADC[nPads];
156  DEId = digits[0].getDetID();
157  double* xPad = getX(xyDxy, nPads);
158  double* yPad = getY(xyDxy, nPads);
159  double* dxPad = getDX(xyDxy, nPads);
160  double* dyPad = getDY(xyDxy, nPads);
161 
162  for (int iDigit = 0; iDigit < digits.size(); ++iDigit) {
163  const auto& digit = digits[iDigit];
164  int padID = digit.getPadID();
165 
166  double x = mSegmentation->padPositionX(padID);
167  double y = mSegmentation->padPositionY(padID);
168  double dx = mSegmentation->padSizeX(padID) / 2.;
169  double dy = mSegmentation->padSizeY(padID) / 2.;
170  uint32_t adc = digit.getADC();
171  float charge(0.);
172  std::memcpy(&charge, &adc, sizeof(adc));
173  bool isSaturated = digit.isSaturated();
174  int plane = mSegmentation->isBendingPad(padID) ? 0 : 1;
175 
176  if (charge <= 0.) {
177  throw std::runtime_error("The precluster contains a digit with charge <= 0");
178  }
179 
180  // GG
181  // Initialisation for GEM processing
182  xPad[iDigit] = x;
183  yPad[iDigit] = y;
184  dxPad[iDigit] = dx;
185  dyPad[iDigit] = dy;
186  padCharge[iDigit] = charge;
187  saturated[iDigit] = isSaturated;
188  cathode[iDigit] = plane;
189  padADC[iDigit] = adc;
190  }
191  // GG Store in dump file
192  uint32_t N = digits.size();
193  // std::cout << " [GEM] Dump PreCluster " << dumpFile->getName() << ", size=" << N << std::endl;
194 
195  if ((N != 0)) {
196  // Replace 0, by 0xFFFFFFFF
197  uint32_t header[6] = {(uint32_t)(bunchCrossing), (orbit), iPreCluster, (0), N, (uint32_t)(DEId)};
198  dumpFile->dumpUInt32(0, 6, header);
199  dumpFile->dumpFloat64(0, N, xPad);
200  dumpFile->dumpFloat64(0, N, yPad);
201  dumpFile->dumpFloat64(0, N, dxPad);
202  dumpFile->dumpFloat64(0, N, dyPad);
203  dumpFile->dumpFloat64(0, N, padCharge);
204  dumpFile->dumpInt16(0, N, saturated);
205  dumpFile->dumpInt16(0, N, cathode);
206  dumpFile->dumpUInt32(0, N, padADC);
207  }
208 }

282  for (int iDigit = 0; iDigit < digits.size(); ++iDigit) {
283  const auto& digit = digits[iDigit];
284  int padID = digit.getPadID();
285 
286  double x = mSegmentation->padPositionX(padID);
287  double y = mSegmentation->padPositionY(padID);
288  double dx = mSegmentation->padSizeX(padID) / 2.;
289  double dy = mSegmentation->padSizeY(padID) / 2.;
290  uint32_t adc = digit.getADC();

315 void ClusterFinderGEM::setClusterResolution(Cluster& cluster) const
316 {
319 
320  if (cluster.getChamberId() < 4) {
321 
322  // do not consider mono-cathode clusters in stations 1 and 2
323  cluster.ex = SDefaultClusterResolution;
324  cluster.ey = SDefaultClusterResolution;
325 
326  } else {
327 
328  // find pads below the cluster
329  int padIDNB(-1), padIDB(-1);
330  bool padsFound = mSegmentation->findPadPairByPosition(cluster.x, cluster.y, padIDB, padIDNB);
331 
332  // look for these pads (if any) in the list of digits associated to this cluster
333  auto itPadNB = mUsedDigits.end();
334  if (padsFound || mSegmentation->isValid(padIDNB)) {
335  itPadNB = std::find_if(mUsedDigits.begin() + cluster.firstDigit, mUsedDigits.end(),
336  [padIDNB](const Digit& digit) { return digit.getPadID() == padIDNB; });
337  }
338  auto itPadB = mUsedDigits.end();
339  if (padsFound || mSegmentation->isValid(padIDB)) {
340  itPadB = std::find_if(mUsedDigits.begin() + cluster.firstDigit, mUsedDigits.end(),
341  [padIDB](const Digit& digit) { return digit.getPadID() == padIDB; });
342  }
343 
344  // set the cluster resolution accordingly
345  cluster.ex = (itPadNB == mUsedDigits.end()) ? SBadClusterResolution : SDefaultClusterResolution;
346  cluster.ey = (itPadB == mUsedDigits.end()) ? SBadClusterResolution : SDefaultClusterResolution;
347  }
348 }

351 void ClusterFinderGEM::findClusters(gsl::span<const Digit> digits,
352  uint16_t bunchCrossing, uint32_t orbit, uint32_t iPreCluster)
353 {
356  // std::cout << " [GEM] preCluster digit size=" << digits.size() << std::endl;
357  // skip preclusters with only 1 digit
358  if (digits.size() < 2) {
359  return;
360  }
361  uint32_t nPreviousCluster = mClusters.size();
362  if (VERBOSE > 0) {
363  printf("----------------------------------------\n");
364  std::cout << " [GEM] PreCluster BC=" << bunchCrossing
365  << ", orbit = " << orbit
366  << "iPreCluster, = " << iPreCluster
367  << std::endl;
368  printf("----------------------------------------\n");
369  }
370  // std::cout << " [GEM] hits/mClusters.size=" << mClusters.size() << std::endl;
371 
372  // set the Mathieson function to be used
373  // GG Inv mMathieson = (digits[0].getDetID() < 300) ? &mMathiesons[0] : &mMathiesons[1];
374 
375  // reset the current precluster being processed
376  // GG Set the Precluster (pads , ....)
377  fillGEMInputData(digits, bunchCrossing, orbit, iPreCluster);
378 
379  // GG process clusters
380  int chId = DEId / 100;
381  int nbrOfHits = clusterProcess(xyDxy, cathode, saturated, padCharge, chId, nPads);
382  double theta[nbrOfHits * 5];
383  Group_t thetaToGroup[nbrOfHits];
384  collectTheta(theta, thetaToGroup, nbrOfHits);
385  // std::cout << " [GEM] Seeds found by GEM " << nbrOfHits << " / nPads = " << nPads << std::endl;
386  double* muX = getMuX(theta, nbrOfHits);
387  double* muY = getMuY(theta, nbrOfHits);
388  double* w = getW(theta, nbrOfHits);
389  // Find subClusters with there seeds
390  //
391  Group_t padToCathGrp[nPads];
392  collectPadToCathGroup(padToCathGrp, nPads);
393  // Take care the number of groups can be !=
394  // between thetaToGroup
395  Group_t nCathGrp = vectorMaxShort(padToCathGrp, nPads);
396  int nPadStored = 0;
397  // Index of the first store digit of the group
398  uint32_t firstDigit;
399  // For all the cath groups
400  for (int g = 1; g < (nCathGrp + 1); g++) {
401  int i = 0;
402  firstDigit = mUsedDigits.size();
403  for (const auto& pad : *mPreCluster) {
404  // Store the pad belonging to the cath-group
405  if (padToCathGrp[i] == g) {
406  mUsedDigits.emplace_back(digits[pad.digitIndex()]);
407  nPadStored++;
408  }
409  i++;
410  }
411  // n digits for the considered group
412  uint32_t nDigits = mUsedDigits.size() - firstDigit;
413  // For all the seeds or hits
414  for (int s = 0; s < nbrOfHits; s++) {
415  // Store the hit belonging to the cath-group
416  if (thetaToGroup[s] == g) {
417  double x = muX[s];
418  double y = muY[s];
419  /*
420  // ??? Do something for the mathieson factors
421  double dx, dy;
422  // ??? value of chID
423  // ??? To do later
424  if (chId <= 4) {
425  dx = SDefaultClusterResolution;
426  dy = SDefaultClusterResolution;
427  } else {
428  // Find the associated pads
429  // set the cluster resolution accordingly
430  // cluster.ex = (NBPad) ? SBadClusterResolution : SDefaultClusterResolution;
431  // cluster.ey = (BPad) ? SBadClusterResolution : SDefaultClusterResolution;
432  // ???
433  dx = SDefaultClusterResolution;
434  dy = SDefaultClusterResolution;
435  }
436  */
437  uint32_t uid = Cluster::buildUniqueId(digits[0].getDetID() / 100 - 1, digits[0].getDetID(), thetaToGroup[s]);
438  mClusters.push_back({
439  static_cast<float>(x), static_cast<float>(y), 0.0, // x, y, z
440  static_cast<float>(0), static_cast<float>(0), // x, y resolution
441  uid, // uid
442  firstDigit, nDigits // firstDigit, nDigits
443  });
444  setClusterResolution(mClusters[mClusters.size() - 1]);
445  }
446  }
447  }
448 
449  // std::cout << " [GEM] Finished preCluster " << digits.size() << std::endl;
450 }
