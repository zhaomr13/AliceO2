40 void BadChannelCalibrationDevice::logStats(size_t dataSize)
41 {
42  static auto loggerStart = std::chrono::high_resolution_clock::now();
43  static auto loggerEnd = loggerStart;
44  static size_t nDigits = 0;
45  static size_t nTF = 0;
46 
47  if (mLoggingInterval == 0) {
48  return;
49  }
50 
51  nDigits += dataSize;
52  nTF += 1;
53 
54  loggerEnd = std::chrono::high_resolution_clock::now();
55  std::chrono::duration<double, std::milli> loggerElapsed = loggerEnd - loggerStart;
56  if (loggerElapsed.count() > 1000) {
57  LOG(info) << "received " << nDigits << " digits in " << nTF << " time frames";
58  nDigits = 0;
59  nTF = 0;
60  loggerStart = std::chrono::high_resolution_clock::now();
61  }
62 }

64 void BadChannelCalibrationDevice::run(o2::framework::ProcessingContext& pc)
65 {
66  const o2::framework::DataProcessingHeader* header = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("digits").header);
67  if (!header) {
68  return;
69  }
70 
71  mTimeStamp = std::min(mTimeStamp, header->creation);
72 
73  auto data = pc.inputs().get<gsl::span<o2::mch::calibration::PedestalDigit>>("digits");
74  mCalibrator->process(header->startTime, data);
75 
76  std::string reason;
77  if (mCalibrator->readyToSend(reason)) {
78  LOGP(info, "We're ready to send output to CCDB ({})", reason);
79  sendOutput(pc.outputs(), reason);
80  }
81  logStats(data.size());
82 }

135 void BadChannelCalibrationDevice::sendOutput(o2::framework::DataAllocator& output,
136  std::string_view reason)
137 {
138  auto& slot = mCalibrator->getFirstSlot();
139  const auto pedData = slot.getContainer();
140  auto nentries = std::accumulate(pedData->cbegin(), pedData->cend(),
141  0,
142  [&](int n, const PedestalChannel& c) { return n + c.mEntries; });
143  auto reason_with_entries = fmt::format("{} ; nentries = {}", reason, nentries);
144 
145  LOGP(info, "sendOutput: {}", reason_with_entries);
146  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
147  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
148 
149  // send regular bad channel object to subspec 0. This regular object
150  // is meant for O2 consumption (in reconstruction and/or simulation)
151  const auto& badChannels = mCalibrator->getBadChannelsVector();
152  auto info = createCcdbInfo(badChannels, mTimeStamp, reason_with_entries);
153 
154  sendCalibrationOutput(output, 0, &badChannels, &info);
155 
156  // send also a simplified (in comma-separated values format) version
157  // of the bad channel list to subspec 1.
158  // This simplified version is meant for DCS usage
159  // (to populate the Oracle online DB for electronics configuration)
160 
161  TObjString badChannelsPOD(toCSV(badChannels).c_str());
162  auto infoPOD = createCcdbInfo(badChannelsPOD, mTimeStamp, reason_with_entries);
163 
164  sendCalibrationOutput(output, 1, &badChannelsPOD, &infoPOD);
165 
166  LOGP(info, "csv={}", badChannelsPOD.String().Data());
167  // and finally send also the data used to compute the bad channel map
168  // on a separate channel (for QC mainly)
169  output.snapshot(o2::framework::Output{"MCH", "PEDESTALS", 0},
170  mCalibrator->getPedestalsVector());
171 }
