56 bool BadChannelCalibrator::hasEnoughData(const Slot& slot) const
57 {
58  const int minNofEntries = BadChannelCalibratorParam::Instance().minRequiredNofEntriesPerChannel;
59  const o2::mch::calibration::PedestalData* pedData = slot.getContainer();
60  auto nofChannels = pedData->size();
61  const int requiredChannels = static_cast<int>(BadChannelCalibratorParam::Instance().minRequiredCalibratedFraction * nofChannels);
62 
63  auto nofCalibrated = std::count_if(pedData->cbegin(), pedData->cend(),
64  [&](const PedestalChannel& c) { return c.mEntries > minNofEntries; });
65 
66  bool hasEnough = nofCalibrated > requiredChannels;
67 
68  LOGP(info,
69  "nofChannelWithEnoughStat(>{})={} nofChannels={} "
70  "requiredChannels={} calibrated={} hasEnough={}",
71  minNofEntries, nofCalibrated, nofChannels, requiredChannels, nofCalibrated, hasEnough);
72 
73  return hasEnough;
74 }

76 void BadChannelCalibrator::finalizeSlot(Slot& slot)
77 {
78  // Extract results for the single slot
79  auto pedestalThreshold = BadChannelCalibratorParam::Instance().maxPed;
80  auto noiseThreshold = BadChannelCalibratorParam::Instance().maxNoise;
81 
82  mPedestalsVector.clear();
83  mBadChannelsVector.clear();
84 
85  o2::mch::calibration::PedestalData* pedestalData = slot.getContainer();
86  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
87 
88  // keep track of first TimeFrame
89  if (slot.getTFStart() < mTFStart) {
90  mTFStart = slot.getTFStart();
91  }
92 
93  for (const auto& ped : *pedestalData) {
94  if (ped.mEntries == 0) {
95  continue;
96  }
97  mPedestalsVector.emplace_back(ped);
98  bool bad = true;
99  if (ped.mPedestal < pedestalThreshold) {
100  if (ped.getRms() < noiseThreshold) {
101  bad = false;
102  }
103  }
104  if (bad) {
105  LOG(info) << ped;
106  mBadChannelsVector.emplace_back(ped.dsChannelId);
107  }
108  }
109 }
