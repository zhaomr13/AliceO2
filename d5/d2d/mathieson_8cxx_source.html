40 void initMathieson()
41 {
42  //
43  for (int i = 0; i < 2; i++) {
44  K2x[i] = M_PI * 0.5 * (1.0 - sqrtK3x[i] * 0.5);
45  K2y[i] = M_PI * 0.5 * (1.0 - sqrtK3y[i] * 0.5);
46  K1x[i] = K2x[i] * sqrtK3x[i] * 0.25 / (atan(sqrtK3x[i]));
47  K1y[i] = K2y[i] * sqrtK3y[i] * 0.25 / (atan(sqrtK3y[i]));
48  K4x[i] = K1x[i] / K2x[i] / sqrtK3x[i];
49  K4y[i] = K1y[i] / K2y[i] / sqrtK3y[i];
50  invPitch[i] = 1.0 / pitch[i];
51  }
52 }

54 void compute2DPadIntegrals(const double* xInf, const double* xSup, const double* yInf, const double* ySup,
55  int N, int chamberId, double Integrals[])
56 {
57  // Returning array: Charge Integral on all the pads
58  //
59  if (chamberId <= 2) {
60  mathiesonType = 0;
61  } else {
62  mathiesonType = 1;
63  }
64  //
65  // Select Mathieson coef.
66  double curK2x = K2x[mathiesonType];
67  double curK2y = K2y[mathiesonType];
68  double curSqrtK3x = sqrtK3x[mathiesonType];
69  double curSqrtK3y = sqrtK3y[mathiesonType];
70  double curK4x = K4x[mathiesonType];
71  double curK4y = K4y[mathiesonType];
72  double curInvPitch = invPitch[mathiesonType];
73  double cst2x = curK2x * curInvPitch;
74  double cst2y = curK2y * curInvPitch;
75  double cst4 = 4.0 * curK4x * curK4y;
76  double uInf, uSup, vInf, vSup;
77 
78  for (int i = 0; i < N; i++) {
79  // x/u
80  uInf = curSqrtK3x * tanh(cst2x * xInf[i]);
81  uSup = curSqrtK3x * tanh(cst2x * xSup[i]);
82  // y/v
83  vInf = curSqrtK3y * tanh(cst2y * yInf[i]);
84  vSup = curSqrtK3y * tanh(cst2y * ySup[i]);
85  //
86  Integrals[i] = cst4 * (atan(uSup) - atan(uInf)) * (atan(vSup) - atan(vInf));
87  // printf(" xyInfSup %2d [%10.6g, %10.6g] x [%10.6g, %10.6g]-> %10.6g\n", i, xInf[i], xSup[i], yInf[i], ySup[i], Integrals[i]);
88  }
89  // printf(" I[0..%3ld] = %f, %f, ... %f\n", N-1, Integrals[0], Integrals[1], Integrals[N-1]);
90  return;
91 }

93 void compute2DMathiesonMixturePadIntegrals(const double* xyInfSup0, const double* theta,
94  int N, int K, int chamberId, double Integrals[])
95 {
96  // Returning array: Charge Integral on all the pads
97  // Remarks:
98  // - This fct is a cumulative one, as a result it should be set to zero
99  // before calling it
100  vectorSetZero(Integrals, N);
101  const double* xInf0 = getConstXInf(xyInfSup0, N);
102  const double* yInf0 = getConstYInf(xyInfSup0, N);
103  const double* xSup0 = getConstXSup(xyInfSup0, N);
104  const double* ySup0 = getConstYSup(xyInfSup0, N);
105  //
106  const double* muX = getConstMuX(theta, K);
107  const double* muY = getConstMuY(theta, K);
108  const double* w = getConstW(theta, K);
109 
110  double z[N];
111  double xyInfSup[4 * N];
112  double* xInf = getXInf(xyInfSup, N);
113  double* yInf = getYInf(xyInfSup, N);
114  double* xSup = getXSup(xyInfSup, N);
115  double* ySup = getYSup(xyInfSup, N);
116  for (int k = 0; k < K; k++) {
117  vectorAddScalar(xInf0, -muX[k], N, xInf);
118  vectorAddScalar(xSup0, -muX[k], N, xSup);
119  vectorAddScalar(yInf0, -muY[k], N, yInf);
120  vectorAddScalar(ySup0, -muY[k], N, ySup);
121  compute2DPadIntegrals(xInf, xSup, yInf, ySup, N, chamberId, z);
122  // printf("Vector Sum %g\n", vectorSum(z, N) );
123  vectorAddVector(Integrals, w[k], z, N, Integrals);
124  }
125 }
