24 bool HitMapBuilder::crossCommonElement(const std::vector<int>& fired, const std::vector<int>& nonFired) const
25 {
26  // First check that all elements in fired are the same
27  auto refIt = fired.begin();
28  for (auto it = fired.begin() + 1, end = fired.end(); it != end; ++it) {
29  if (*it != *refIt) {
30  return false;
31  }
32  }
33 
34  // If there is no element in nonFired, then return true
35  if (nonFired.empty()) {
36  return true;
37  }
38 
39  // Check that at least one element in nonFired matches the reference elements
40  for (auto it = nonFired.begin(), end = nonFired.end(); it != end; ++it) {
41  if (*it == *refIt) {
42  return true;
43  }
44  }
45 
46  return false;
47 }

66 void HitMapBuilder::buildTrackInfo(Track& track, gsl::span<const Cluster> clusters) const
67 {
68  std::vector<int> firedLocIds, nonFiredLocIds;
69  std::vector<int> firedDeIds, firedRPCLines, nonFiredRPCLines;
70  bool outsideAcceptance = false;
71  for (int ich = 0; ich < 4; ++ich) {
72  auto icl = track.getClusterMatched(ich);
73  if (icl >= 0) {
74  auto& cl = clusters[icl];
75  firedDeIds.emplace_back(cl.deId);
76  firedRPCLines.emplace_back(detparams::getRPCLine(cl.deId));
77  firedLocIds.emplace_back(getLocId(cl.xCoor, cl.yCoor, cl.deId));
78  // auto localPt = mHitFinder.getGeometryTransformer().globalToLocal(cl.deId, cl.xCoor, cl.yCoor, cl.zCoor);
79  for (int icath = 0; icath < 2; ++icath) {
80  if (cl.isFired(icath)) {
81  track.setFiredChamber(ich, icath);
82  }
83  }
84  } else {
85  auto impactPts = mHitFinder.getLocalPositions(track, ich);
86  for (auto& impactPt : impactPts) {
87  nonFiredRPCLines.emplace_back(detparams::getRPCLine(impactPt.deId));
88  nonFiredLocIds.emplace_back(getLocId(impactPt.xCoor, impactPt.yCoor, impactPt.deId));
89  if (nonFiredLocIds.back() == 0) {
90  outsideAcceptance = true;
91  }
92  }
93  }
94  }
95  track.setFiredDeId(firedDeIds.front());
96  track.setFiredLocalBoard(firedLocIds.front());
97  int effFlag = 0;
98  if (!outsideAcceptance) {
99  effFlag = getEffFlag(firedRPCLines, nonFiredRPCLines, firedLocIds, nonFiredLocIds);
100  }
101  track.setEfficiencyFlag(effFlag);
102 }
