43 {
44  const Mapper& mapper = Mapper::instance();
45  o2::utils::TreeStreamRedirector pcstream(outFileName, "RECREATE");
46  pcstream.GetFile()->cd();
47 
48  if (integrationIntervals <= 0) {
49  integrationIntervals = static_cast<int>(getNIntegrationIntervals(mCRUs.front()));
50  }
51 
52  std::vector<float> idcOneA = mIDCOne.mIDCOne[0];
53  std::vector<float> idcOneC = mIDCOne.mIDCOne[1];
54  for (unsigned int integrationInterval = 0; integrationInterval < integrationIntervals; ++integrationInterval) {
55  const unsigned int nIDCsSector = Mapper::getPadsInSector() * Mapper::NSECTORS;

63  std::vector<float> idcsZero(nIDCsSector);
64  std::vector<float> idcsDelta(nIDCsSector);

67  std::vector<unsigned int> sectorv(nIDCsSector);
68

131 void o2::tpc::IDCFactorization::calcIDCZero(const bool norm)
132 {
133  const unsigned int nIDCsSide = mNIDCsPerSector * o2::tpc::SECTORSPERSIDE;
134  mIDCZero.clear();
135  mIDCZero.resize(nIDCsSide);
136 
137 #pragma omp parallel for num_threads(sNThreads)
138  for (unsigned int cruInd = 0; cruInd < mCRUs.size(); ++cruInd) {
139  const unsigned int cru = mCRUs[cruInd];
140  const o2::tpc::CRU cruTmp(cru);
141  const auto side = cruTmp.side();
142  const unsigned int region = cruTmp.region();
143  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * (cruTmp.sector() % o2::tpc::SECTORSPERSIDE);
144  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
145  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
146  const unsigned int indexGlob = (idcs % mNIDCsPerCRU[region]) + factorIndexGlob;
147  if (norm && mIDCs[cru][timeframe][idcs] > 0) {
148  mIDCs[cru][timeframe][idcs] *= Mapper::INVPADAREA[region];
149  }
150  mIDCZero.fillValueIDCZero(mIDCs[cru][timeframe][idcs], side, indexGlob);
151  }
152  }
153  }
154  std::transform(mIDCZero.mIDCZero[Side::A].begin(), mIDCZero.mIDCZero[Side::A].end(), mIDCZero.mIDCZero[Side::A].begin(), [normVal = getNIntegrationIntervals(mCRUs.front())](auto& val) { return val / normVal; });
155  std::transform(mIDCZero.mIDCZero[Side::C].begin(), mIDCZero.mIDCZero[Side::C].end(), mIDCZero.mIDCZero[Side::C].begin(), [normVal = getNIntegrationIntervals(mCRUs.front())](auto& val) { return val / normVal; });
156 }

158 void o2::tpc::IDCFactorization::fillIDCZeroDeadPads()
159 {
160 #pragma omp parallel for num_threads(sNThreads)
161  for (unsigned int cruInd = 0; cruInd < mCRUs.size(); ++cruInd) {
162  const unsigned int cru = mCRUs[cruInd];
163  const o2::tpc::CRU cruTmp(cru);
164  const Side side = cruTmp.side();
165  const int region = cruTmp.region();
166  const int sector = cruTmp.sector();
167  std::vector<unsigned int> padTmp;
168  padTmp.reserve(Mapper::PADSPERROW[region].back());
169  for (unsigned int lrow = 0; lrow < Mapper::ROWSPERREGION[region]; ++lrow) {
170  float idcRow = 0;
171  int count = 0;
172  const unsigned int integrationInterval = 0;
173  // loop over pad row and calculate mean of IDC0
174  for (unsigned int pad = 0; pad < Mapper::PADSPERROW[region][lrow]; ++pad) {
175  const auto index = getIndexUngrouped(sector, region, lrow, pad, integrationInterval);
176  const auto idcZeroVal = mIDCZero.mIDCZero[side][index];
177  if (idcZeroVal > 0) {
178  const auto globalPad = Mapper::getGlobalPadNumber(lrow, pad, region);
179  const float gain = (mGainMap ? mGainMap->getValue(sector, globalPad) : 1);
180  idcRow += idcZeroVal / gain;
181  ++count;
182  } else {
183  padTmp.emplace_back(pad);
184  }
185  }
186  // loop over dead pads and set value
187  for (const auto pad : padTmp) {
188  const float meanIDC0 = idcRow / count;
189  const auto globalPad = Mapper::getGlobalPadNumber(lrow, pad, region);
190  const float gain = (mGainMap ? mGainMap->getValue(sector, globalPad) : 1);
191  const float idcZero = gain * meanIDC0;
192  const auto index = getIndexUngrouped(sector, region, lrow, pad, integrationInterval);
193  mIDCZero.setValueIDCZero(idcZero, side, index);
194  }
195  padTmp.clear();
196  }
197  }

200 void o2::tpc::IDCFactorization::calcIDCOne()
201 {
202  const unsigned int integrationIntervals = getNIntegrationIntervals(mCRUs.front());
203  mIDCOne.clear();
204  mIDCOne.resize(integrationIntervals);
205  const unsigned int crusPerSide = Mapper::NREGIONS * SECTORSPERSIDE;
206 
207  unsigned int crusPerSideA = 0;
208  unsigned int crusPerSideC = 0;
209  for (unsigned int cru = 0; cru < crusPerSide; ++cru) {
210  if (mIDCs[cru].front().size() > 0) {
211  ++crusPerSideA;
212  }
213  if (mIDCs[cru + crusPerSide].front().size() > 0) {
214  ++crusPerSideC;
215  }
216  }
217 
218  std::array<std::vector<std::vector<float>>, SIDES> idcOneSafe;
219  for (auto& vecSide : idcOneSafe) {
220  vecSide.resize(sNThreads);
221  for (auto& interavalvec : vecSide) {
222  interavalvec.resize(integrationIntervals);
223  }
224  }
225 
226 #pragma omp parallel for num_threads(sNThreads)
227  for (unsigned int cruInd = 0; cruInd < mCRUs.size(); ++cruInd) {
228  const unsigned int cru = mCRUs[cruInd];
229  const o2::tpc::CRU cruTmp(cru);
230  const unsigned int region = cruTmp.region();
231  const auto side = cruTmp.side();
232  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * (cruTmp.sector() % o2::tpc::SECTORSPERSIDE);
233  unsigned int integrationIntervallast = 0;
234  std::vector<unsigned short> count(integrationIntervals);
235  std::vector<float> idcOneTmp(integrationIntervals);
236  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
237  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
238  const unsigned int integrationInterval = idcs / mNIDCsPerCRU[region] + integrationIntervallast;
239  const unsigned int localPad = (idcs % mNIDCsPerCRU[region]);
240  const unsigned int indexGlob = localPad + factorIndexGlob;
241  const auto idcZeroVal = mIDCZero.mIDCZero[side][indexGlob];
242 
243  // check pad in case of input is not grouped
244  if (!mInputGrouped && mPadFlagsMap) {
245  const o2::tpc::PadFlags flag = mPadFlagsMap->getCalArray(cru).getValue(localPad);
246  if ((flag & PadFlags::flagSkip) == PadFlags::flagSkip) {
247  continue;
248  }
249  }
250 
251  if (idcZeroVal > 0 && mIDCs[cru][timeframe][idcs] > 0) {
252  idcOneTmp[integrationInterval] += mIDCs[cru][timeframe][idcs] / idcZeroVal;
253  ++count[integrationInterval];
254  }
255  }
256  integrationIntervallast += mIDCs[cru][timeframe].size() / mNIDCsPerCRU[region];
257  }
258  const float crusTmp = (side == Side::A) ? crusPerSideA : crusPerSideC;
259 #ifdef WITH_OPENMP
260  const int ithread = omp_get_thread_num();
261 #else
262  const int ithread = 0;
263 #endif
264  for (int i = 0; i < integrationIntervals; ++i) {
265  if (count[i] == 0) {
266  continue;
267  }
268  idcOneSafe[side][ithread][i] += idcOneTmp[i] / (crusTmp * count[i]);
269  }
270  }
271 
272  for (int side = 0; side < SIDES; ++side) {
273  for (const auto& interavalvec : idcOneSafe[side]) {
274  std::transform(mIDCOne.mIDCOne[side].begin(), mIDCOne.mIDCOne[side].end(), interavalvec.begin(), mIDCOne.mIDCOne[side].begin(), std::plus<float>());
275  }
276  }
277 }

279 void o2::tpc::IDCFactorization::calcIDCDelta()
280 {
281  const unsigned int nIDCsSide = mNIDCsPerSector * SECTORSPERSIDE;
282  for (unsigned int i = 0; i < getNChunks(); ++i) {
283  const auto idcsSide = nIDCsSide * getNIntegrationIntervals(i, mCRUs.front());
284  mIDCDelta[i].getIDCDelta(Side::A).resize(idcsSide);
285  mIDCDelta[i].getIDCDelta(Side::C).resize(idcsSide);
286  }
287 
288 #pragma omp parallel for num_threads(sNThreads)
289  for (unsigned int cruInd = 0; cruInd < mCRUs.size(); ++cruInd) {
290  const unsigned int cru = mCRUs[cruInd];
291  const o2::tpc::CRU cruTmp(cru);
292  const unsigned int region = cruTmp.region();
293  const auto side = cruTmp.side();
294  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * (cruTmp.sector() % o2::tpc::SECTORSPERSIDE);
295  unsigned int integrationIntervallast = 0;
296  unsigned int integrationIntervallastLocal = 0;
297  unsigned int lastChunk = 0;
298 
299  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
300  const unsigned int chunk = getChunk(timeframe);
301  if (lastChunk != chunk) {
302  integrationIntervallastLocal = 0;
303  }
304 
305  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
306  const unsigned int intervallocal = idcs / mNIDCsPerCRU[region];
307  const unsigned int integrationIntervalGlobal = intervallocal + integrationIntervallast;
308  const unsigned int integrationIntervalLocal = intervallocal + integrationIntervallastLocal;
309  const unsigned int localPad = idcs % mNIDCsPerCRU[region];
310  const unsigned int indexGlob = localPad + factorIndexGlob;
311  const auto idcZero = mIDCZero.mIDCZero[side][indexGlob];
312  const auto idcOne = mIDCOne.mIDCOne[side][integrationIntervalGlobal];
313  const auto mult = idcZero * idcOne;
314  auto val = (mult > 0 && mIDCs[cru][timeframe][idcs] > 0) ? mIDCs[cru][timeframe][idcs] / mult - 1 : 0;
315  if (!mInputGrouped && mPadFlagsMap) {
316  const o2::tpc::PadFlags flag = mPadFlagsMap->getCalArray(cru).getValue(localPad);
317  if ((flag & PadFlags::flagSkip) == PadFlags::flagSkip) {
318  val = 0;
319  }
320  }
321  mIDCDelta[chunk].setIDCDelta(side, indexGlob + integrationIntervalLocal * nIDCsSide, val);
322  }
323 
324  const unsigned int intervals = mIDCs[cru][timeframe].size() / mNIDCsPerCRU[region];
325  integrationIntervallast += intervals;
326  integrationIntervallastLocal += intervals;
327  lastChunk = chunk;
328  }
329  }

342 void o2::tpc::IDCFactorization::createStatusMap()
343 {
344  const static auto& paramIDCGroup = ParameterIDCGroup::Instance();
345  mPadFlagsMap = std::make_unique<CalDet<PadFlags>>(CalDet<PadFlags>("flags", PadSubset::Region));
346 #pragma omp parallel for num_threads(sNThreads)
347  for (unsigned int cruInd = 0; cruInd < mCRUs.size(); ++cruInd) {
348  const unsigned int cru = mCRUs[cruInd];
349  const o2::tpc::CRU cruTmp(cru);
350  const Side side = cruTmp.side();
351  const int region = cruTmp.region();
352  const int sector = cruTmp.sector();
353  std::vector<float> idcsRow;
354  const auto maxValues = Mapper::PADSPERROW[region].back();
355  idcsRow.reserve(maxValues);
356  for (unsigned int lrow = 0; lrow < Mapper::ROWSPERREGION[region]; ++lrow) {
357 
358  // loop over pads in row in the first iteration and calculate median at the end
359  // in the second iteration check if the IDC value is to far away from the median
360  for (int iter = 0; iter < 2; ++iter) {
361  const unsigned int integrationInterval = 0;
362  const float median = (iter == 1) ? getMedian(idcsRow) : 0;
363  // loop over pad row and calculate mean of IDC0
364  for (unsigned int pad = 0; pad < Mapper::PADSPERROW[region][lrow]; ++pad) {
365  const auto index = getIndexUngrouped(sector, region, lrow, pad, integrationInterval);
366  const auto idcZeroVal = mIDCZero.mIDCZero[side][index];
367 
368  if (iter == 0) {
369  // exclude dead pads
370  if (idcZeroVal > 0) {
371  idcsRow.emplace_back(idcZeroVal);
372  }
373  } else {
374  if (idcZeroVal <= 0) {
375  const unsigned int padInRegion = Mapper::OFFSETCRULOCAL[region][lrow] + pad;
376  const o2::tpc::PadFlags flag = o2::tpc::PadFlags::flagDeadPad | o2::tpc::PadFlags::flagSkip | mPadFlagsMap->getCalArray(cru).getValue(padInRegion);
377  mPadFlagsMap->getCalArray(cru).setValue(padInRegion, flag);
378  } else if (idcZeroVal > paramIDCGroup.maxIDC0Median * median) {
379  const unsigned int padInRegion = Mapper::OFFSETCRULOCAL[region][lrow] + pad;
380  const o2::tpc::PadFlags flag = o2::tpc::PadFlags::flagHighPad | o2::tpc::PadFlags::flagSkip | mPadFlagsMap->getCalArray(cru).getValue(padInRegion);
381  mPadFlagsMap->getCalArray(cru).setValue(padInRegion, flag);
382  } else if (idcZeroVal < paramIDCGroup.minIDC0Median * median) {
383  const unsigned int padInRegion = Mapper::OFFSETCRULOCAL[region][lrow] + pad;
384  const o2::tpc::PadFlags flag = o2::tpc::PadFlags::flagLowPad | o2::tpc::PadFlags::flagSkip | mPadFlagsMap->getCalArray(cru).getValue(padInRegion);
385  mPadFlagsMap->getCalArray(cru).setValue(padInRegion, flag);
386  }
387  }
388  } // loop over pads in row
389  } // end iteration
390  idcsRow.clear();
391  }
392  }

457 void o2::tpc::IDCFactorization::getLocalIntegrationInterval(const unsigned int region, const unsigned int integrationInterval, unsigned int& chunk, unsigned int& localintegrationInterval) const
458 {
459  unsigned int nintervals = 0;
460  unsigned int nitervalsChunk = 0;
461  unsigned int globalTF = 0;
462  for (unsigned int ichunk = 0; ichunk < getNChunks(); ++ichunk) {
463  const auto nTFsPerChunk = getNTFsPerChunk(ichunk);
464  for (unsigned int tf = 0; tf < nTFsPerChunk; ++tf) {
465  nintervals += mIDCs[region][globalTF].size() / mNIDCsPerCRU[region];
466  if (integrationInterval < nintervals) {
467  chunk = getChunk(globalTF);
468  localintegrationInterval = integrationInterval - nitervalsChunk;
469  return;
470  }
471  ++globalTF;
472  }
473  nitervalsChunk = nintervals;
474  }
