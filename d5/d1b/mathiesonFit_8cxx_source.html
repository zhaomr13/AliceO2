22 int f_ChargeIntegral(const gsl_vector* gslParams, void* dataFit, gsl_vector* residuals)
23 {
24  funcDescription_t* dataPtr = (funcDescription_t*)dataFit;
25  int N = dataPtr->N;
26  int K = dataPtr->K;
27  double* x = dataPtr->x_ptr;
28  double* y = dataPtr->y_ptr;
29  double* dx = dataPtr->dx_ptr;
30  double* dy = dataPtr->dy_ptr;
31  Mask_t* cath = dataPtr->cath_ptr;
32  double* zObs = dataPtr->zObs_ptr;
33  Mask_t* notSaturated = dataPtr->notSaturated_ptr;
34  int chamberId = dataPtr->chamberId;
35  double* cathWeights = dataPtr->cathWeights_ptr;
36  double* cathMax = dataPtr->cathMax_ptr;
37  double* zCathTotalCharge = dataPtr->zCathTotalCharge_ptr;
38  int verbose = dataPtr->verbose;
39  // Parameters
40  const double* params = gsl_vector_const_ptr(gslParams, 0);
41  // Note:
42  // mux = mu[0:K-1]
43  // muy = mu[K:2K-1]
44  const double* mu = &params[0];
45  double* w = (double*)&params[2 * K];
46 
47  // Set constrain: sum_(w_k) = 1
48  // Rewriten ??? w[K-1] = 1.0 - vectorSum( w, K-1 );
49  double lastW = 1.0 - vectorSum(w, K - 1);
50  if (verbose > 1) {
51  printf(" Function evaluation at:\n");
52  for (int k = 0; k < K; k++) {
53  printf(" mu_k[%d] = %g %g \n", k, mu[k], mu[K + k]);
54  }
55  for (int k = 0; k < K - 1; k++) {
56  printf(" w_k[%d] = %g \n", k, w[k]);
57  }
58  // Last W
59  printf(" w_k[%d] = %g \n", K - 1, lastW);
60  }
61 
62  // Charge Integral on Pads
63  double z[N];
64  // Not used
65  double z_k[N];
66  double zTmp[N];
67  // TODO: optimize compute before
68  double xyInfSup[4 * N];
69  //
70  double* xInf = getXInf(xyInfSup, N);
71  double* xSup = getXSup(xyInfSup, N);
72  double* yInf = getYInf(xyInfSup, N);
73  double* ySup = getYSup(xyInfSup, N);
74 
75  vectorSetZero(z, N);
76  for (int k = 0; k < K; k++) {
77  // xInf[:] = x[:] - dx[:] - muX[k]
78  vectorAddVector(x, -1.0, dx, N, xInf);
79  vectorAddScalar(xInf, -mu[k], N, xInf);
80  // xSup = xInf + 2.0 * dxy[0]
81  vectorAddVector(xInf, 2.0, dx, N, xSup);
82  // yInf = xy[1] - dxy[1] - mu[k,1]
83  // ySup = yInf + 2.0 * dxy[1]
84  vectorAddVector(y, -1.0, dy, N, yInf);
85  vectorAddScalar(yInf, -mu[K + k], N, yInf);
86  // ySup = yInf + 2.0 * dxy[0]
87  vectorAddVector(yInf, 2.0, dy, N, ySup);
88  //
89  // z[:] += w[k] * cathWeight
90  // * computeMathieson2DIntegral( xInf[:], xSup[:], yInf[:], ySup[:], N )
91  // Inv. ??? compute2DPadIntegrals(xyInfSup, N, chamberId, zTmp);
92  compute2DPadIntegrals(xInf, xSup, yInf, ySup, N, chamberId, zTmp);
93  // GG Note: if require compute the charge of k-th component
94  // z_k[k] = vectorSum( zTmp, N )
95  // Done after the loop: vectorMultVector( zTmp, cathWeights, N, zTmp);
96  double wTmp = (k != K - 1) ? w[k] : lastW;
97  // saturated pads
98  // ??? ?
99  // vectorMaskedMult( zTmp, notSaturated, N, zTmp);
100  vectorAddVector(z, wTmp, zTmp, N, z);
101  }
102  // Normalize each cathode with unsaturated pads
103  // Not used ???
104  double sumNormalizedZ[2];
105  for (int i = 0; i < N; i++) {
106  if (cath[i] == 0) {
107  sumNormalizedZ[0] += notSaturated[i] * z[i];
108  } else {
109  sumNormalizedZ[1] += notSaturated[i] * z[i];
110  }
111  }
112 
113  // Normalize with the max
114  // TODO ??? Can be done in fitMatiesson
115  double maxThZ[2] = {0, 0};
116  for (int i = 0; i < N; i++) {
117  maxThZ[cath[i]] = fmax(maxThZ[cath[i]], notSaturated[i] * z[i]);
118  }
119  // Avoid dividing by 0
120  for (int c = 0; c < 2; c++) {
121  if (maxThZ[c] < 1.0e-6) {
122  // cathMax[c] sould be 0
123  maxThZ[c] = 1.0;
124  }
125  }
126  // Normalization coef
127  double coefNorm[2] = {cathMax[0] / maxThZ[0], cathMax[1] / maxThZ[1]};
128  double meanCoef = (coefNorm[0] + coefNorm[1]) / ((coefNorm[0] > 1.0e-6) + (coefNorm[1] > 1.0e-6));
129  //
130  for (int i = 0; i < N; i++) {
131  z[i] = z[i] * coefNorm[cath[i]];
132  }
133 
134  if (verbose > 1) {
135  printf(" Max of unsaturated (observed) pads (cathMax0/1)= %f, %f, maxThZ (computed) %f, %f\n", cathMax[0], cathMax[1], maxThZ[0], maxThZ[1]);
136  }
137  // double cathPenal = fabs(zCathTotalCharge[0] - zCath0) + fabs(zCathTotalCharge[1] - zCath1);
138  double cathPenal = 0;
139  // ??? vectorAdd( zObs, -1.0, residual );
140  // TODO Optimize (elementwise not a good solution)
141  double wPenal = 0.0;
142  // ??? why abs value for penalties
143  for (int k = 0; k < (K - 1); k++) {
144  if (w[k] < 0.0) {
145  wPenal += (-w[k]);
146  } else if (w[k] > 1.0) {
147  wPenal += (w[k] - 1.0);
148  }
149  }
150  // printf(" coefNorm ??? %f, %f\n", maxThZ[0], maxThZ[1]);
151  // printf(" coefNorm ??? %f, %f\n", coefNorm[0], coefNorm[1]);
152  // printf(" meanCoef ??? %f\n", meanCoef);
153  wPenal = wPenal + fabs(1.0 - vectorSum(w, K - 1) - lastW);
154  if (verbose > 1) {
155  printf(" wPenal: %f\n", wPenal);
156  }
157  for (int i = 0; i < N; i++) {
158  // gsl_vector_set(residuals, i, (zObs[i] - z[i]) * (1.0 + cathPenal) + wPenal);
159  double mask = notSaturated[i];
160  // if ((notSaturated[i] == 0) && (z[i] < zObs[i]) && (z[i] > 1.5 * zObs[i]) ) {
161  if ((notSaturated[i] == 0) && (z[i] < zObs[i])) {
162  // mask = 1.0;
163  // Test
164  // Don't consider saturated pads
165  // mask = notSaturated[i];
166  mask = 1.0;
167  }
168  //
169  // mask = 1.0;
170  gsl_vector_set(residuals, i, mask * ((zObs[i] - z[i]) + meanCoef * wPenal));
171  // gsl_vector_set(residuals, i, mask * (zObs[i] - z[i]) + 0 * wPenal);
172  }
173  if (verbose > 1) {
174  printf(" Observed sumCath0=%15.8f, sumCath1=%15.8f,\n", zCathTotalCharge[0], zCathTotalCharge[1]);
175  // printf(" fitted sumCath0=%15.8f, sumCath1=%15.8f,\n", zCath0, zCath1);
176  printf(" Penalties cathPenal=%5.4g wPenal=%5.4g \n", 1.0 + cathPenal, wPenal);
177  printf(" Residues\n");
178  printf(" %15s %15s %15s %15s %15s %15s\n", "zObs", "z", "cathWeight", "norm. factor", "notSaturated", "residual");
179  for (int i = 0; i < N; i++) {
180  printf(" %15.8f %15.8f %15.8f %15.8f %d %15.8f\n",
181  zObs[i], z[i], cathWeights[i], sumNormalizedZ[cath[i]] * cathWeights[i], notSaturated[i], gsl_vector_get(residuals, i));
182  }
183  printf("\n");
184  }
185  return GSL_SUCCESS;
186 }

400  p = p >> 1;

444  }
445  } else {
446  // Function description (extra data nor parameters)
447  mathiesonData.N = N;
448  mathiesonData.K = KMax;
449  mathiesonData.x_ptr = getX(xyDxy, N);
450  mathiesonData.y_ptr = getY(xyDxy, N);
451  mathiesonData.dx_ptr = getDX(xyDxy, N);

453  mathiesonData.cath_ptr = cath;
454  mathiesonData.zObs_ptr = z;
455  mathiesonData.notSaturated_ptr = notSaturated;
456  // Init the weights

460  cathMax[cath[i]] = fmax(cathMax[cath[i]], notSaturated[i] * z[i]);

462  }
463

465  mathiesonData.cathMax_ptr = cathMax;
466  mathiesonData.chamberId = chamberId;
467  mathiesonData.zCathTotalCharge_ptr = zCathTotalCharge;
468  mathiesonData.verbose = verbose;

477  f.params = &mathiesonData;
478 
479  bool doFit = true;
480  // K test
481  int K = KMax;
482  // Sort w
483  int maxIndex[KMax];
484  for (int k = 0; k < KMax; k++) {
485  maxIndex[k] = k;
486  }
487  double* w = &muAndWi[2 * KMax];
488  std::sort(maxIndex, &maxIndex[KMax], [=](int a, int b) { return (w[a] > w[b]); });
489 
490  while (doFit) {
491  // Select the best K's
492  // Copy kTest max
493  double muAndWTest[3 * K];
494  // Mu part
495  for (int k = 0; k < K; k++) {
496  // Respecttively mux, muy, w
497  muAndWTest[k] = muAndWi[maxIndex[k]];
498  muAndWTest[k + K] = muAndWi[maxIndex[k] + KMax];
499  muAndWTest[k + 2 * K] = muAndWi[maxIndex[k] + 2 * KMax];
500  }

502  vectorPrint(" Selected w", &muAndWTest[2 * K], K);

504  vectorPrint(" Selected muy", &muAndWTest[K], K);
