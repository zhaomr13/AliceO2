50 o2::zdc::Digitizer::BCCache& DigitizerTest::getCreateBCCache(const o2::InteractionRecord& ir)
51 {
52  // printf("size = %zu Add %u.%u\n", mCache.size(), ir.orbit, ir.bc);
53  if (mCache.empty() || mCache.back() < ir) {
54  mCache.emplace_back();
55  auto& cb = mCache.back();
56  cb = ir;
57  return cb;
58  }
59  if (mCache.front() > ir) {
60  mCache.emplace_front();
61  auto& cb = mCache.front();
62  cb = ir;
63  return cb;
64  }
65  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
66  if ((*cb) == ir) {
67  return *cb;
68  }
69  if (ir < (*cb)) {
70  auto cbnew = mCache.emplace(cb); // insert new element before cb
71  (*cbnew) = ir;
72  return (*cbnew);
73  }
74  }
75  return mCache.front();
76 }

79 double DigitizerTest::add(int ic, float myAmp, const o2::InteractionRecord irpk,
80  float myShift, bool hasJitter)
81 {
82  // myShift is the position of the signal peak inside the bunch crossing identified by irpk (in ns)
83  // and is converted into sample units
84  myShift = myShift * NTimeBinsPerBC / o2::constants::lhc::LHCBunchSpacingNS;
85  // Shift of samples due to additional fluctuations
86  float shift = hasJitter ? gRandom->Gaus(0, mSimCondition->channels[ic].pedestalNoise) : 0;
87  int pos_min = mSimCondition->channels[ic].ampMinID;
88  int nbx = mSimCondition->channels[ic].shape.size();
89  // This may result into a wrap of the BC counter. However the signal is stored only
90  // in pre-allocated containers
91  o2::InteractionRecord ir = irpk + BCCacheMin;
92  for (Int_t ib = 0; ib < NBC2Cache; ib++) {
93  auto* bc = getBCCache(ir);
94  if (bc) {
95  // Flag the presence of channel data
96  auto& bcd = getCreateBCData(ir);
97  bcd.channels = bcd.channels | mMask[ic];
98  auto sbcd = NTimeBinsPerBC * ir.differenceInBC(irpk);
99  int sample = TMath::Nint(pos_min + TSN * (sbcd - myShift) + shift + TSNH);
100  for (int i = 0; i < NTimeBinsPerBC; i++) {
101  if (sample >= 0 && sample < nbx) {
102  double y = mSimCondition->channels[ic].shape[sample];
103  bc->data[ic][i] += y * myAmp;
104  // LOG(info) << ic << " " << ir.orbit << "." << ir.bc << " s" << i << ") s " << sample << " " << y*myAmp << " -> " << bc->data[ic][i];
105  }
106  sample += TSN;
107  }
108  }
109  ir++;
110  }
111  // Assigned position in units of bunch crossing
112  return irpk.orbit * o2::constants::lhc::LHCMaxBunches + irpk.bc + (myShift + shift / double(TSN)) / double(NTimeBinsPerBC);
113 } // add

116 void DigitizerTest::digitize()
117 {
118  for (auto bc = mCache.begin(); bc != mCache.end(); bc++) {
119  if (zdcOrbitData.empty() || bc->orbit != zdcOrbitData.back().ir.orbit) {
120  auto& od = zdcOrbitData.emplace_back();
121  od.ir.orbit = bc->orbit;
122  od.ir.bc = o2::constants::lhc::LHCMaxBunches - 1;
123  // Rough estimate of pedestal fluctuations to fill orbit data
124  for (int ic = 0; ic < NChannels; ic++) {
125  // 20 is a guess (imagine that pedestal is computed with an average over 400 empty bunch crossings)
126  od.data[ic] = TMath::Nint(FOffset *
127  gRandom->Gaus(mSimCondition->channels[ic].pedestal,
128  mSimCondition->channels[ic].pedestalFluct / 20.));
129  }
130  // printf("Adding data for orbit=%u\n", od.ir.orbit);
131  }
132  // printf("Adding CH data for bunch=%u\n", bc->bc);
133  auto& bcd = getCreateBCData(*bc);
134  bcd.ref.setFirstEntry(zdcChData.size());
135  uint32_t nstored = 0;
136  for (int ic = 0; ic < NChannels; ic++) {
137  // Check if channel has data for this bunch crossing
138  if (bcd.channels & mMask[ic]) {
139  Double_t meanp = mSimCondition->channels[ic].pedestal;
140  Double_t sigmab = mSimCondition->channels[ic].pedestalFluct;
141  // LOG(info) << ic << " meanp=" << meanp << " sigmab=" << sigmab;
142  // No pedestal oscillations on data. We do it in the orbit data
143  for (int i = 0; i < NTimeBinsPerBC; i++) {
144  Double_t yval = TMath::Nint(bc->data[ic][i] + gRandom->Gaus(meanp, sigmab));
145  if (yval > ADCMax) {
146  yval = ADCMax;
147  } else if (yval < ADCMin) {
148  yval = ADCMin;
149  }
150  // This is not correct but will work for direct digitization
151  bc->digi[ic][i] = yval;
152  // LOG(info) << ic << " " << i << ") " << yval;
153  }
154  // Store data in output array
155  zdcChData.emplace_back(ic, bc->digi[ic]);
156  nstored++;
157  }
158  }
159  bcd.ref.setEntries(nstored);
160  }
161 } // digitize
