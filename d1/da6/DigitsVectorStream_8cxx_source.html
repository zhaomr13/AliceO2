36 void DigitsVectorStream::addNoiseDigits(LabeledDigit& d1)
37 {
38  double amplitude = d1.getAmplitude();
39  double sigma = mSimParam->getPinNoise();
40  if (amplitude > constants::EMCAL_HGLGTRANSITION * constants::EMCAL_ADCENERGY) {
41  sigma = mSimParam->getPinNoiseLG();
42  }
43 
44  double noise = std::abs(mRandomGenerator->Gaus(0, sigma));
45  MCLabel label(true, 1.0);
46  LabeledDigit d(d1.getTower(), noise, d1.getTimeStamp(), label);
47  d1 += d;
48 }

51 void DigitsVectorStream::fill(gsl::span<o2::emcal::DigitTimebin> digitlist, o2::InteractionRecord record)
52 {
53  std::map<unsigned int, std::list<LabeledDigit>> outputList;
54 
55  for (auto& digitsTimeBin : digitlist) {
56 
57  for (auto& [tower, digitsList] : *digitsTimeBin.mDigitMap) {
58 
59  if (digitsList.size() == 0) {
60  continue;
61  }
62  digitsList.sort();
63 
64  for (auto& ld : digitsList) {
65 
66  // Loop over all digits in the time sample and sum the digits that belongs to the same tower and falls in one time bin
67  for (auto ld1 = digitsList.begin(); ld1 != digitsList.end(); ++ld1) {
68 
69  if (ld == *ld1) {
70  continue;
71  }
72 
73  std::vector<decltype(digitsList.begin())> toDelete;
74 
75  if (ld.canAdd(*ld1)) {
76  ld += *ld1;
77  toDelete.push_back(ld1);
78  }
79  for (auto del : toDelete) {
80  digitsList.erase(del);
81  }
82  }
83 
84  if (mSimulateNoiseDigits) {
85  addNoiseDigits(ld);
86  }
87 
88  if (mRemoveDigitsBelowThreshold && (ld.getAmplitude() < (mSimParam->getDigitThreshold() * constants::EMCAL_ADCENERGY))) {
89  continue;
90  }
91  if (ld.getAmplitude() < 0) {
92  continue;
93  }
94  if ((ld.getTimeStamp() >= mSimParam->getLiveTime()) || (ld.getTimeStamp() < 0)) {
95  continue;
96  }
97 
98  outputList[tower].push_back(ld);
99  }
100  }
101  }
102 
103  for (const auto& [tower, outdiglist] : outputList) {
104  for (const auto& d : outdiglist) {
105  // outdiglist.sort();
106 
107  Digit digit = d.getDigit();
108  std::vector<MCLabel> labels = d.getLabels();
109  mDigits.push_back(digit);
110 
111  Int_t LabelIndex = mLabels.getIndexedSize();
112  for (const auto& label : labels) {
113  mLabels.addElementRandomAccess(LabelIndex, label);
114  }
115  }
116  }
117 
118  mTriggerRecords.emplace_back(record, o2::trigger::PhT, mStartIndex, mDigits.size());
119  mStartIndex = mDigits.size();
120 
121  LOG(info) << "Have " << mStartIndex << " digits ";
122 }
