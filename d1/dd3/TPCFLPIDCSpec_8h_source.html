90  void init(o2::framework::InitContext& ic) final
91  {
92  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
93  if (mLoadFromFile) {
94  const char* fName = "IDCGroup.root";
95  TFile fInp(fName, "READ");
96  for (TObject* keyAsObj : *fInp.GetListOfKeys()) {
97  const auto key = dynamic_cast<TKey*>(keyAsObj);
98  const char* name = key->GetName();
99  LOGP(info, "Key name: {} Type: {}", name, key->GetClassName());
100 
101  if (std::strcmp(o2::tpc::IDCAverageGroup<IDCAverageGroupCRU>::Class()->GetName(), key->GetClassName()) != 0) {
102  LOGP(info, "skipping object. wrong class.");
103  continue;
104  }
105  IDCAverageGroup<IDCAverageGroupCRU>* idcavg = (IDCAverageGroup<IDCAverageGroupCRU>*)fInp.Get(name);
106  unsigned int cru = idcavg->getSector() * Mapper::NREGIONS + idcavg->getRegion();
107  // check cru
108  if (std::find(mCRUs.begin(), mCRUs.end(), cru) != mCRUs.end()) {
109  mIDCStruct.mIDCs[cru].processIDCs(mPadFlagsMap);
110  mIDCStruct.mIDCs[cru].setFromFile(fName, name);
111  }
112  delete idcavg;
113  }
114  }

118  void run(o2::framework::ProcessingContext& pc) final
119  {
120  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
121  LOGP(info, "averaging and grouping IDCs for TF {} for CRUs {} to {} using {} threads", processing_helpers::getCurrentTF(pc), mCRUs.front(), mCRUs.back(), mIDCStruct.mIDCs.begin()->second.getNThreads());
122  } else {
123  LOGP(info, "skipping grouping of IDCs for TF {} for CRUs {} to {}", processing_helpers::getCurrentTF(pc), mCRUs.front(), mCRUs.back());
124  }
125 
126  // retrieving map containing the status flags for the static outlier
127  if (mLoadPadMapCCDB) {
128  updateTimeDependentParams(pc);
129  }
130 
131  if (!mLoadFromFile) {
132  for (int i = 0; i < mCRUs.size() + mLoadPadMapCCDB; ++i) {
133  const DataRef ref = pc.inputs().getByPos(i);
134  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
135  const auto descr = tpcCRUHeader->dataDescription;
136  if (TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim) == descr) {
137  const int cru = tpcCRUHeader->subSpecification >> 7;
138  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
139  mIDCStruct.mIDCs[cru].setIDCs(pc.inputs().get<std::vector<float>>(ref));
140  mIDCStruct.mIDCs[cru].processIDCs(mPadFlagsMap);
141  } else {
142  mIDCStruct.mIDCs[cru] = pc.inputs().get<std::vector<float>>(ref);
143  }
144  // send the output for one CRU for one TF
145  sendOutput(pc.outputs(), cru);
146  }
147  }
148  } else {
149  for (int i = 0; i < mCRUs.size(); ++i) {
150  sendOutput(pc.outputs(), mCRUs[i]);
151  }
152  }
153 
154  if (mDebug) {
155  TFile fOut(fmt::format("IDCGroup_{}_tf_{}.root", mLane, processing_helpers::getCurrentTF(pc)).data(), "RECREATE");
156  for (int i = 0; i < mCRUs.size(); ++i) {
157  const DataRef ref = pc.inputs().getByPos(i);
158  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
159  const int cru = tpcCRUHeader->subSpecification >> 7;
160  fOut.WriteObject(&mIDCStruct.mIDCs[cru], fmt::format("CRU_{}", cru).data());
161  }
162  }
163  }

210  void sendOutput(DataAllocator& output, const uint32_t cru)
211  {
212  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
213 
214  std::vector<float> vec1DIDCs;
215  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
216  // calculate 1D-IDCs = sum over 2D-IDCs as a function of the integration interval
217  vec1DIDCs = mIDCStruct.mIDCs[cru].getIDCGroup().get1DIDCs();
218  } else {
219  vec1DIDCs = IDCGroup::get1DIDCsUngrouped(mIDCStruct.mIDCs[cru], cru, mPadFlagsMap);
220  }
221  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDC(), subSpec, Lifetime::Timeframe}, vec1DIDCs);
222 
223  // TODO use this and fix #include <boost/container/pmr/polymorphic_allocator.hpp> in ROOT CINT
224  // output.adoptContainer(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, std::move(mIDCs[cru]).getIDCGroupData());
225  output.snapshot(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, mIDCStruct.getData(cru));
226 
227  mBuffer1DIDCs[cru].emplace_back(std::move(vec1DIDCs));
228  mBuffer1DIDCs[cru].pop_front(); // removing oldest 1D-IDCs
229 
230  fill1DIDCs(cru);
231  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDCEPN(), subSpec, Lifetime::Timeframe}, mOneDIDCs);
232  }

234  void fill1DIDCs(const uint32_t cru)
235  {
236  // fill 1D-IDC vector with mRangeIDC buffered values
237  unsigned int i = mRangeIDC;
238  for (auto it = mBuffer1DIDCs[cru].crbegin(); it != mBuffer1DIDCs[cru].crend(); ++it) {
239  for (auto idc = it->crbegin(); idc != it->crend(); ++idc) {
240  mOneDIDCs[--i] = *idc;
241  if (i == 0) {
242  return;
243  }
244  }
245  }

249 template <class Type>
250 DataProcessorSpec getTPCFLPIDCSpec(const int ilane, const std::vector<uint32_t>& crus, const unsigned int rangeIDC, const bool debug, const bool loadFromFile, const bool loadStatusMap = false)
251 {
252  std::vector<OutputSpec> outputSpecs;
253  std::vector<InputSpec> inputSpecs;
254  outputSpecs.reserve(crus.size());
255  inputSpecs.reserve(crus.size());
256 
257  for (const auto& cru : crus) {
258  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
259  if (!loadFromFile) {
260  inputSpecs.emplace_back(InputSpec{"idcs", gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), subSpec, Lifetime::Timeframe});
261  }
262 
263  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescriptionIDCGroup(), subSpec});
264  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescription1DIDC(), subSpec});
265  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescription1DIDCEPN(), subSpec});
266  }
267 
268  if (loadStatusMap) {
269  LOGP(info, "Using pad status map from CCDB");
270  inputSpecs.emplace_back("tpcpadmap", gDataOriginTPC, "PADSTATUSMAP", 0, Lifetime::Condition, ccdbParamSpec("TPC/Calib/IDC/PadStatusMap"));
271  }
272 
273  const auto id = fmt::format("tpc-flp-idc-{:02}", ilane);
274  return DataProcessorSpec{
275  id.data(),
276  inputSpecs,
277  outputSpecs,
278  AlgorithmSpec{adaptFromTask<TPCFLPIDCDevice<Type>>(ilane, crus, rangeIDC, debug, loadFromFile, loadStatusMap)}}; // end DataProcessorSpec
279 }
