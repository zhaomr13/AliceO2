96 template <typename T>
97 class PedestalDataIterator
98 {
99  public:
100  using value_type = T;
101  using difference_type = std::ptrdiff_t;
102  using pointer = T*;
103  using reference = T&;
104  using iterator_category = std::forward_iterator_tag;
105 
106  PedestalDataIterator() = default;
107  explicit PedestalDataIterator(PedestalData* data) : mData{data},
108  mMapIt{},
109  mCol{0},
110  mRow{0}
111  {
112  if (mData && mData->size() == 0) {
113  mData = nullptr;
114  }
115  if (mData) {
116  mMapIt = mData->mPedestals.begin();
117  }
118  }
119 
120  reference operator*() const
121  {
122  return mMapIt->second[mRow][mCol];
123  }
124  pointer operator->() const
125  {
126  return &mMapIt->second[mRow][mCol];
127  }
128 
129  PedestalDataIterator& operator++()
130  {
131  if (mMapIt != mData->mPedestals.end()) {
132  if (mCol < PedestalData::MAXCHANNEL - 1) {
133  mCol++;
134  return *this;
135  }
136  if (mRow < PedestalData::MAXDS - 1) {
137  mCol = 0;
138  mRow++;
139  return *this;
140  }
141  ++mMapIt;
142  mCol = 0;
143  mRow = 0;
144  if (mMapIt != mData->mPedestals.end()) {
145  return *this;
146  }
147  mData = nullptr;
148  }
149  // undefined behavior here (should not increment an end iterator)
150  return *this;
151  }
152 
153  bool operator==(const PedestalDataIterator& rhs)
154  {
155  return mData == rhs.mData &&
156  mMapIt == rhs.mMapIt &&
157  mCol == rhs.mCol &&
158  mRow == rhs.mRow;
159  }
160 
161  bool operator!=(const PedestalDataIterator& rhs) { return !(*this == rhs); }
162 
163  PedestalDataIterator operator++(int)
164  {
165  auto tmp = *this;
166  ++(*this);
167  return tmp;
168  }
169 
170  private:
171  PedestalData* mData;
172  PedestalData::PedestalsMap::iterator mMapIt;
173  int mCol;
174  int mRow;
175 };
