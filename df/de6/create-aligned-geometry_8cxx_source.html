24 void createAlignedGeometry(long timeToPick, std::string ccdbServer = "")
25 {
26  if (timeToPick == 0) {
27  timeToPick = o2::ccdb::getCurrentTimestamp();
28  }
29  auto& cm = o2::ccdb::BasicCCDBManager::instance();
30  if (!ccdbServer.empty()) {
31  cm.setURL(ccdbServer);
32  }
33  cm.setTimestamp(timeToPick);
34  cm.get<TGeoManager>("GLO/Config/Geometry");
35  if (!o2::base::GeometryManager::isGeometryLoaded()) {
36  throw std::runtime_error(fmt::format("Failed to load geometry from {} for timestamp {}", cm.getURL(), timeToPick));
37  }
38  gGeoManager->SetName(std::string(o2::base::NameConf::CCDBOBJECT).c_str());
39  o2::base::GeometryManager::applyMisalignent();
40  auto fnm = o2::base::NameConf::getAlignedGeomFileName();
41  gGeoManager->Export(fnm.c_str());
42  LOG(info) << "Stored to local file " << fnm;
43 }
