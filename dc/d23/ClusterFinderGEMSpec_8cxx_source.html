161  void run(framework::ProcessingContext& pc)
162  {
164 
165  // get the input preclusters and associated digits
166  auto preClusterROFs = pc.inputs().get<gsl::span<ROFRecord>>("preclusterrofs");
167  auto preClusters = pc.inputs().get<gsl::span<PreCluster>>("preclusters");
168  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
169 
170  // LOG(info) << "received time frame with " << preClusterROFs.size() << " interactions";
171 
172  // create the output messages for clusters and attached digits
173  auto& clusterROFs = pc.outputs().make<std::vector<ROFRecord>>(OutputRef{"clusterrofs"});
174  auto& clusters = pc.outputs().make<std::vector<Cluster>>(OutputRef{"clusters"});
175  auto& usedDigits = pc.outputs().make<std::vector<Digit>>(OutputRef{"clusterdigits"});
176 
177  clusterROFs.reserve(preClusterROFs.size());
178  for (const auto& preClusterROF : preClusterROFs) {
179  LOG(info) << "processing interaction: time frame " << preClusterROF.getBCData().orbit << "...";
180  // GG infos
181  // uint16_t bc = DummyBC; ///< bunch crossing ID of interaction
182  // uint32_t orbit = DummyOrbit; ///< LHC orbit
183  // clusterize every preclusters
184  uint16_t bCrossing = preClusterROF.getBCData().bc;
185  uint32_t orbit = preClusterROF.getBCData().orbit;
186  uint32_t iPreCluster = 0;
187  auto tStart = std::chrono::high_resolution_clock::now();
188  // Inv ??? if ( orbit==22 ) {
189  //
190  if (isOriginalActivated()) {
191  mClusterFinderOriginal.reset();
192  }
193  if (isGEMActivated()) {
194  mClusterFinderGEM.reset();
195  }
196  // Get the starting index for new cluster founds
197  size_t startGEMIdx = mClusterFinderGEM.getClusters().size();
198  size_t startOriginalIdx = mClusterFinderOriginal.getClusters().size();
199  // std::cout << "Start index GEM=" << startGEMIdx << ", Original=" << startOriginalIdx << std::endl;
200  for (const auto& preCluster : preClusters.subspan(preClusterROF.getFirstIdx(), preClusterROF.getNEntries())) {
201  // Inv ??? for (const auto& preCluster : preClusters.subspan(preClusterROF.getFirstIdx(), 1102)) {
202  startGEMIdx = mClusterFinderGEM.getClusters().size();
203  startOriginalIdx = mClusterFinderOriginal.getClusters().size();
204  // Dump preclusters
205  // std::cout << "bCrossing=" << bCrossing << ", orbit=" << orbit << ", iPrecluster" << iPreCluster
206  // << ", PreCluster: digit start=" << preCluster.firstDigit <<" , digit size=" << preCluster.nDigits << std::endl;
207  if (isOriginalDumped()) {
208  mClusterFinderGEM.dumpPreCluster(mOriginalDump, digits.subspan(preCluster.firstDigit, preCluster.nDigits), bCrossing, orbit, iPreCluster);
209  }
210  if (isGEMDumped()) {
211  mClusterFinderGEM.dumpPreCluster(mGEMDump, digits.subspan(preCluster.firstDigit, preCluster.nDigits), bCrossing, orbit, iPreCluster);
212  }
213  // Clusterize
214  if (isOriginalActivated()) {
215  mClusterFinderOriginal.findClusters(digits.subspan(preCluster.firstDigit, preCluster.nDigits));
216  }
217  if (isGEMActivated()) {
218  mClusterFinderGEM.findClusters(digits.subspan(preCluster.firstDigit, preCluster.nDigits), bCrossing, orbit, iPreCluster);
219  }
220  // Dump clusters (results)
221  // std::cout << "[Original] total clusters.size=" << mClusterFinderOriginal.getClusters().size() << std::endl;
222  // std::cout << "[GEM ] total clusters.size=" << mClusterFinderGEM.getClusters().size() << std::endl;
223  if (isOriginalDumped()) {
224  mClusterFinderGEM.dumpClusterResults(mOriginalDump, mClusterFinderOriginal.getClusters(), startOriginalIdx, bCrossing, orbit, iPreCluster);
225  }
226  if (isGEMDumped()) {
227  mClusterFinderGEM.dumpClusterResults(mGEMDump, mClusterFinderGEM.getClusters(), startGEMIdx, bCrossing, orbit, iPreCluster);
228  }
229  // if ( isGEMDumped())
230  iPreCluster++;
231  }
232  // } // Inv ??? if ( orbit==22 ) {
233  auto tEnd = std::chrono::high_resolution_clock::now();
234  mTimeClusterFinder += tEnd - tStart;
235 
236  // fill the ouput messages
237  if (isGEMOutputStream()) {
238  clusterROFs.emplace_back(preClusterROF.getBCData(), clusters.size(), mClusterFinderGEM.getClusters().size());
239  } else {
240  clusterROFs.emplace_back(preClusterROF.getBCData(), clusters.size(), mClusterFinderOriginal.getClusters().size());
241  }
242  //
243  writeClusters(clusters, usedDigits);
244  }
245 
246  LOGP(info, "Found {:4d} clusters from {:4d} preclusters in {:2d} ROFs",
247  clusters.size(), preClusters.size(), preClusterROFs.size());

252  void writeClusters(std::vector<Cluster, o2::pmr::polymorphic_allocator<Cluster>>& clusters,
253  std::vector<Digit, o2::pmr::polymorphic_allocator<Digit>>& usedDigits) const
254  {
257  auto clusterOffset = clusters.size();
258  if (isGEMOutputStream()) {
259  clusters.insert(clusters.end(), mClusterFinderGEM.getClusters().begin(), mClusterFinderGEM.getClusters().end());
260  } else {
261  clusters.insert(clusters.end(), mClusterFinderOriginal.getClusters().begin(), mClusterFinderOriginal.getClusters().end());
262  }
263  auto digitOffset = usedDigits.size();
264  if (isGEMOutputStream()) {
265  usedDigits.insert(usedDigits.end(), mClusterFinderGEM.getUsedDigits().begin(), mClusterFinderGEM.getUsedDigits().end());
266  } else {
267  usedDigits.insert(usedDigits.end(), mClusterFinderOriginal.getUsedDigits().begin(), mClusterFinderOriginal.getUsedDigits().end());
268  }
269 
270  for (auto itCluster = clusters.begin() + clusterOffset; itCluster < clusters.end(); ++itCluster) {
271  itCluster->firstDigit += digitOffset;
272  }
273  }
