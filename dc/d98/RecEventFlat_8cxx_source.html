56 int RecEventFlat::next()
57 {
58  if (mEntry >= mNEntries) {
59  return 0;
60  }
61  // Reconstruction messages
62  clearBitmaps();
63  // Channel data
64  ezdc.clear();
65  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
66  TDCVal[itdc].clear();
67  TDCAmp[itdc].clear();
68  }
69 
70  // Get References
71  mCurB = mRecBC->at(mEntry);
72  mCurB.getRef(mFirstE, mNE, mFirstT, mNT, mFirstI, mNI);
73  mStopE = mFirstE + mNE;
74  mStopT = mFirstT + mNT;
75  mStopI = mFirstI + mNI;
76 
77  ir = mCurB.ir;
78  channels = mCurB.channels;
79  triggers = mCurB.triggers;
80 
81  // Decode event info
82  mDecodedInfo.clear();
83  int infoState = 0;
84  uint16_t code = 0;
85  uint32_t map = 0;
86  for (int i = mFirstI; i < mStopI; i++) {
87  uint16_t info = mInfo->at(i);
88  if (infoState == 0) {
89  if (info & 0x8000) {
90  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
91  break;
92  }
93  code = info & 0x03ff;
94  uint8_t ch = (info >> 10) & 0x1f;
95  if (ch == 0x1f) {
96  infoState = 1;
97  } else if (ch < NChannels) {
98  decodeInfo(ch, code);
99  } else {
100  LOGF(error, "Info about non existing channel: %u", ch);
101  }
102  } else if (infoState == 1) {
103  if (info & 0x8000) {
104  map = info & 0x7fff;
105  } else {
106  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
107  break;
108  }
109  infoState = 2;
110  } else if (infoState == 2) {
111  if (info & 0x8000) {
112  uint32_t maph = info & 0x7fff;
113  map = (maph << 15) | map;
114  decodeMapInfo(map, code);
115  } else {
116  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
117  break;
118  }
119  infoState = 0;
120  }
121  }
122 
123  // Decode energy
124  for (int i = mFirstE; i < mStopE; i++) {
125  auto myenergy = mEnergy->at(i);
126  auto ch = myenergy.ch();
127  ezdc[ch] = myenergy.energy();
128  // Assign implicit event info
129  if (adcPedOr[ch] == false && adcPedQC[ch] == false && adcPedMissing[ch] == false) {
130  adcPedEv[ch] = true;
131  }
132  }
133 
134  // Decode TDCs
135  for (int i = mFirstT; i < mStopT; i++) {
136  auto mytdc = mTDCData->at(i);
137  auto ch = mytdc.ch();
138  if (ch < NTDCChannels) {
139  if (mytdc.isBeg()) {
140  isBeg[ch] = true;
141  }
142  if (mytdc.isEnd()) {
143  isEnd[ch] = true;
144  }
145  TDCVal[ch].push_back(mytdc.val);
146  TDCAmp[ch].push_back(mytdc.amp);
147  // Assign implicit event info
148  if (tdcPedQC[ch] == false && tdcPedMissing[ch] == false) {
149  tdcPedOr[ch] = true;
150  }
151  }
152  }
153 
154  mEntry++;
155  return mEntry;
156 }

158 void RecEventFlat::decodeMapInfo(uint32_t map, uint16_t code)
159 {
160 #ifdef O2_ZDC_DEBUG
161  printf("decodeMapInfo%08x code=%u\n", map, code);
162 #endif
163  for (uint8_t ch = 0; ch < NChannels; ch++) {
164  if (map & (0x1 << ch)) {
165  decodeInfo(ch, code);
166  }

170 void RecEventFlat::decodeInfo(uint8_t ch, uint16_t code)
171 {
172  if (mVerbosity != DbgZero) {
173  printf("%9u.%04u Info: ch=%2d (%s) code=%-4u (%s)\n", ir.orbit, ir.bc, ch, ch < NChannels ? ChannelNames[ch].data() : "N.D.",
174  code, code < MsgEnd ? MsgText[code].data() : "undefined");
175  }
176  // Reconstruction messages
177  switch (code) {
178  case MsgGeneric:
179  genericE[ch] = true;
180  break;
181  case MsgTDCPedQC:
182  tdcPedQC[ch] = true;
183  break;
184  case MsgTDCPedMissing:
185  tdcPedMissing[ch] = true;
186  break;
187  case MsgADCPedOr:
188  adcPedOr[ch] = true;
189  break;
190  case MsgADCPedQC:
191  adcPedQC[ch] = true;
192  break;
193  case MsgADCPedMissing:
194  adcPedMissing[ch] = true;
195  break;
196  case MsgOffPed:
197  offPed[ch] = true;
198  break;
199  case MsgPilePed:
200  pilePed[ch] = true;
201  break;
202  case MsgPileTM:
203  pileTM[ch] = true;
204  break;
205  case MsgADCMissingwTDC:
206  adcMissingwTDC[ch] = true;
207  break;
208  case MsgTDCPileEvC:
209  tdcPileEvC[ch] = true;
210  break;
211  case MsgTDCPileEvE:
212  tdcPileEvE[ch] = true;
213  break;
214  case MsgTDCPileM1C:
215  tdcPileM1C[ch] = true;
216  break;
217  case MsgTDCPileM1E:
218  tdcPileM1E[ch] = true;
219  break;
220  case MsgTDCPileM2C:
221  tdcPileM2C[ch] = true;
222  break;
223  case MsgTDCPileM2E:
224  tdcPileM2E[ch] = true;
225  break;
226  case MsgTDCPileM3C:
227  tdcPileM3C[ch] = true;
228  break;
229  case MsgTDCPileM3E:
230  tdcPileM3E[ch] = true;
231  break;
232  case MsgTDCSigE:
233  tdcSigE[ch] = true;
234  break;
235  // End_of_messages
236  default:
237  LOG(error) << "Not managed info code: " << code;
238  return;
239  }
240  mDecodedInfo.emplace_back((code & 0x03ff) | ((ch & 0x1f) << 10));
241 }

243 void RecEventFlat::print() const
244 {
245  printf("%9u.%04u ", ir.orbit, ir.bc);
246  printf("nE %2d pos %d nT %2d pos %d nI %2d pos %d\n", mNE, mFirstE, mNT, mFirstT, mNI, mFirstI);
247  printf("%9u.%04u ", ir.orbit, ir.bc);
248  printf("Read:");
249  for (int ic = 0; ic < NDigiChannels; ic++) {
250  if (ic % NChPerModule == 0) {
251  if (ic == 0) {
252  printf(" %d[", ic / NChPerModule);
253  } else {
254  printf("] %d[", ic / NChPerModule);
255  }
256  }
257  if (channels & (0x1 << ic)) {
258  printf("R");
259  } else {
260  printf(" ");
261  }
262  }
263  printf("]\n");
264  printf("%9u.%04u ", ir.orbit, ir.bc);
265  printf("Hits:");
266  for (int ic = 0; ic < NDigiChannels; ic++) {
267  if (ic % NChPerModule == 0) {
268  if (ic == 0) {
269  printf(" %d[", ic / NChPerModule);
270  } else {
271  printf("] %d[", ic / NChPerModule);
272  }
273  }
274  bool is_hit = triggers & (0x1 << ic);
275  bool is_trig = mTriggerMask & (0x1 << ic);
276  if (is_trig) {
277  if (is_hit) {
278  printf("T");
279  } else {
280  printf(".");
281  }
282  } else {
283  if (is_hit) {
284  printf("H");
285  } else {
286  printf(" ");
287  }
288  }
289  }
290  printf("]\n");
291 }

293 void RecEventFlat::printDecodedMessages() const
294 {
295  const std::array<bool, NChannels>* maps[MsgEnd];
296  maps[0] = &genericE;
297  maps[1] = &tdcPedQC;
298  maps[2] = &tdcPedMissing;
299  maps[3] = &adcPedOr;
300  maps[4] = &adcPedQC;
301  maps[5] = &adcPedMissing;
302  maps[6] = &offPed;
303  maps[7] = &pilePed;
304  maps[8] = &pileTM;
305  maps[9] = &adcMissingwTDC;
306  maps[10] = &tdcPileEvC;
307  maps[11] = &tdcPileEvE;
308  maps[12] = &tdcPileM1C;
309  maps[13] = &tdcPileM1E;
310  maps[14] = &tdcPileM2C;
311  maps[15] = &tdcPileM2E;
312  maps[16] = &tdcPileM3C;
313  maps[17] = &tdcPileM3E;
314  maps[18] = &tdcSigE;
315  // End_of_messages
316 
317  for (int32_t imsg = 0; imsg < MsgEnd; imsg++) {
318  TString msg = TString::Format("%-30s:", MsgText[imsg].data());
319  if (maps[imsg] == nullptr) {
320  continue;
321  }
322  bool found = false;
323  for (int32_t isig = 0; isig < NChannels; isig++) {
324  if (maps[imsg]->at(isig) == true) {
325  found = true;
326  msg += TString::Format(" %s", ChannelNames[isig].data());
327  } else {
328  msg += " ";
329  }
330  }
331  if (found) {
332  printf("%u.%04u Info %s\n", ir.orbit, ir.bc, msg.Data());
333  }
334  }
