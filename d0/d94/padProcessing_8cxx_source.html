272 int checkConsistencyMapKToIJ(const PadIdx_t* aloneIPads, const PadIdx_t* aloneJPads, int N0, int N1)
273 {
274  MapKToIJ_t ij;
275  int n = 0;
276  int rc = 0;
277  // Consistency with intersectionMatrix
278  // and aloneI/JPads
279  for (PadIdx_t k = 0; k < nbrOfProjPads; k++) {
280  ij = mapKToIJ[k];
281  if ((ij.i >= 0) && (ij.j >= 0)) {
282  if (intersectionMatrix[ij.i * N1 + ij.j] != 1) {
283  printf("ERROR: no intersection %d %d %d\n", ij.i, ij.j, intersectionMatrix[ij.i * N1 + ij.j]);
284  throw std::overflow_error("Divide by zero exception");
285  rc = -1;
286  } else {
287  n++;
288  }
289  } else if (ij.i < 0) {
290  if (aloneJPads[ij.j] != k) {
291  printf("ERROR: j-pad should be alone %d %d %d %d\n", ij.i, ij.j, aloneIPads[ij.j], k);
292  throw std::overflow_error("Divide by zero exception");
293  rc = -1;
294  }
295  } else if (ij.j < 0) {
296  if (aloneIPads[ij.i] != k) {
297  printf("ERROR: i-pad should be alone %d %d %d %d\n", ij.i, ij.j, aloneJPads[ij.j], k);
298  throw std::overflow_error("Divide by zero exception");
299  rc = -1;
300  }
301  }
302  }
303  // TODO : Make a test with alone pads ???
304  int sum = vectorSumInt(intersectionMatrix, N0 * N1);
305  if (sum != n) {
306  printf("ERROR: nbr of intersection differs %d %d \n", n, sum);
307  throw std::overflow_error("Divide by zero exception");
308  rc = -1;
309  }
310 
311  for (int i = 0; i < N0; i++) {
312  for (int j = 0; j < N1; j++) {
313  int k = mapIJToK[i * N1 + j];
314  if (k >= 0) {
315  ij = mapKToIJ[k];
316  if ((ij.i != i) || (ij.j != j)) {
317  throw std::overflow_error("checkConsistencyMapKToIJ: MapIJToK/MapKToIJ");
318  }
319  }
320  }
321  }
322  // Check mapKToIJ / mapIJToK
323  for (PadIdx_t k = 0; k < nbrOfProjPads; k++) {
324  ij = mapKToIJ[k];
325  if (ij.i < 0) {
326  if (aloneJPads[ij.j] != k) {
327  printf("i, j, k = %d, %d %d\n", ij.i, ij.j, k);
328  throw std::overflow_error("checkConsistencyMapKToIJ: MapKToIJ/MapIJToK aloneJPads");
329  }
330  } else if (ij.j < 0) {
331  if (aloneIPads[ij.i] != k) {
332  printf("i, j, k = %d, %d %d\n", ij.i, ij.j, k);
333  throw std::overflow_error("checkConsistencyMapKToIJ: MapKToIJ/MapIJToK aloneIPads");
334  }
335  } else if (mapIJToK[ij.i * N1 + ij.j] != k) {
336  printf("i, j, k = %d, %d %d\n", ij.i, ij.j, k);

341  return rc;
