100  void run(o2::framework::ProcessingContext& pc) final
101  {
102  // set the min range of TFs for first TF
103  if (mProcessedTFs == 0) {
104  mTFFirst = processing_helpers::getCurrentTF(pc);
105  mTimeStampFirst = processing_helpers::getTimeStamp(pc) / 1000; // in milliseconds
106 
107  // write struct containing grouping parameters to access grouped IDCs to CCDB
108  if (mWriteToDB && mUpdateGroupingPar) {
109  // validity for grouping parameters is from first TF to some really large TF (until it is updated) TODO do somewhere else?!
110  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
111  mDBapi.storeAsTFileAny<o2::tpc::ParameterIDCGroupCCDB>(&mIDCStruct.mIDCs.getIDCGroupHelperSector().getGroupingParameter(), "TPC/Calib/IDC/GROUPINGPAR", mMetadata, mTimeStampFirst, 99999999999999);
112  } else {
113  mDBapi.storeAsTFileAny<o2::tpc::ParameterIDCGroupCCDB>(&mIDCFactorization.getGroupingParameter(), "TPC/Calib/IDC/GROUPINGPAR", mMetadata, mTimeStampFirst, 99999999999999);
114  }
115  mUpdateGroupingPar = false; // write grouping parameters only once
116  }
117 
118  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
119  mTFRangeIDCDelta[iChunk] = getFirstTFDeltaIDC(iChunk);
120  }
121  }
122 
123  // check if current TF is in range of IDCDelta range
124  findTimeStamp(pc);
125 
126  for (int i = 0; i < mCRUs.size() + 1; ++i) {
127  const DataRef ref = pc.inputs().getByPos(i);
128  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
129  const auto descr = tpcCRUHeader->dataDescription;
130  if (TPCDistributeIDCSpec::getDataDescriptionIDC() == descr) {
131  const int cru = tpcCRUHeader->subSpecification - mLaneId * CRU::MaxCRU;
132  mIDCFactorization.setIDCs(pc.inputs().get<std::vector<float>>(ref), cru, mProcessedTFs); // aggregate IDCs
133  }
134  }
135  ++mProcessedTFs;
136 
137  LOGP(info, "aggregated TFs: {}", mProcessedTFs);
138 
139  if (mProcessedTFs == mIDCFactorization.getNTimeframes()) {
140  mProcessedTFs = 0; // reset processed TFs for next aggregation interval
141  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
142  mIDCFactorization.factorizeIDCs(true); // calculate DeltaIDC, 0D-IDC, 1D-IDC
143  } else {
144  mIDCFactorization.factorizeIDCs(false); // calculate DeltaIDC, 0D-IDC, 1D-IDC
145  }
146 
147  if (mDebug) {
148  LOGP(info, "dumping aggregated and factorized IDCs to file");
149  mIDCFactorization.dumpToFile(fmt::format("IDCFactorized_{:02}.root", processing_helpers::getCurrentTF(pc)).data());
150  mIDCFactorization.dumpPadFlagMap("padstatusmap.root", "PadStatus");
151  }
152 
153  // storing to CCDB
154  sendOutput(pc.outputs());
155  }
156  }

227  void sendOutput(DataAllocator& output)
228  {
229  if (mSendOutDebug) {
230  sendOutputDebug(output);
231  }
232 
233  if (mWriteToDB) {
234  const auto timeStampStart = mTimeStampFirst;
235  const auto timeStampEnd = 99999999999999;
236 
237  LOGP(info, "Writing IDCs to CCDB");
238  mDBapi.storeAsTFileAny<o2::tpc::IDCZero>(&mIDCFactorization.getIDCZero(), "TPC/Calib/IDC/IDC0", mMetadata, timeStampStart, timeStampEnd);
239  mDBapi.storeAsTFileAny<o2::tpc::IDCOne>(&mIDCFactorization.getIDCOne(), "TPC/Calib/IDC/IDC1", mMetadata, timeStampStart, timeStampEnd);
240 
241  auto padStatusMap = mIDCFactorization.getPadStatusMap();
242  if (padStatusMap) {
243  // store map in case it is nullptr
244  if (!mPadFlagsMap) {
245  mPadFlagsMap = std::move(padStatusMap);
246  LOGP(info, "Writing pad status map to CCDB.");
247  mDBapi.storeAsTFileAny<CalDet<PadFlags>>(mPadFlagsMap.get(), "TPC/Calib/IDC/PadStatusMap", mMetadata, timeStampStart, timeStampEnd);
248  LOGP(info, "Pad status map written to CCDB");
249  } else {
250  // check if map changed. if it changed update the map in the CCDB and store new map in buffer
251  if (!(*padStatusMap.get() == *mPadFlagsMap.get())) {
252  LOGP(info, "Pad status map changed");
253  LOGP(info, "Writing pad status map to CCDB");
254  mDBapi.storeAsTFileAny<CalDet<PadFlags>>(mPadFlagsMap.get(), "TPC/Calib/IDC/PadStatusMap", mMetadata, timeStampStart, timeStampEnd);
255  LOGP(info, "Pad status map written to CCDB");
256  }
257  }
258  }
259 
260  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
261  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
262  // perform grouping of IDC Delta if necessary
263  mIDCStruct.mIDCs.setIDCs(std::move(mIDCFactorization).getIDCDeltaUncompressed(iChunk));
264  LOGP(info, "averaging and grouping DeltaIDCs for TFs {} - {} for CRUs {} to {} using {} threads", getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk), mCRUs.front(), mCRUs.back(), mIDCStruct.mIDCs.getNThreads());
265  mIDCStruct.mIDCs.processIDCs(mPadFlagsMap.get());
266  if (mDebug) {
267  mIDCStruct.mIDCs.dumpToFile(fmt::format("IDCDeltaAveraged_chunk{:02}_{:02}.root", iChunk, getFirstTFDeltaIDC(iChunk)).data());
268  }
269  }
270 
271  switch (mCompressionDeltaIDC) {
272  case IDCDeltaCompression::MEDIUM:
273  default: {
274  using compType = unsigned short;
275  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
276  auto idcDeltaMediumCompressed = IDCDeltaCompressionHelper<compType>::getCompressedIDCs(mIDCStruct.mIDCs.getIDCGroupData());
277  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
278  } else {
279  auto idcDeltaMediumCompressed = mIDCFactorization.getIDCDeltaMediumCompressed(iChunk);
280  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
281  }
282 
283  break;
284  }
285  case IDCDeltaCompression::HIGH: {
286  using compType = unsigned char;
287  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
288  auto idcDeltaMediumCompressed = IDCDeltaCompressionHelper<compType>::getCompressedIDCs(mIDCStruct.mIDCs.getIDCGroupData());
289  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
290  } else {
291  auto idcDeltaHighCompressed = mIDCFactorization.getIDCDeltaHighCompressed(iChunk);
292  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaHighCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
293  }
294  break;
295  }
296  case IDCDeltaCompression::NO:
297  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
298  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCStruct.mIDCs.getIDCGroupData(), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
299  } else {
300  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCFactorization.getIDCDeltaUncompressed(iChunk), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTimeStampDeltaIDC(iChunk), timeStampEnd);
301  }
302  break;
303  }
304  }
305  }
306  // reseting aggregated IDCs. This is done for safety, but if all data is received in the next aggregation interval it isnt necessary... remove it?
307  mIDCFactorization.reset();
308  }
