48  void init(o2::framework::InitContext& ic)
49  {
50 
51  if (!gGeoManager) {
52  o2::base::GeometryManager::loadGeometry();
53  }
54 
55  auto geoTrans = createTransformationFromManager(gGeoManager);
56 
57  mTracker = std::make_unique<Tracker>(geoTrans);
58 
59  mHitMapBuilder = std::make_unique<HitMapBuilder>(geoTrans);
60 
61  if (!mTracker->init(true)) {
62  LOG(error) << "Initialization of MID tracker device failed";
63  }
64 
65  auto stop = [this]() {
66  LOG(info) << "Capacities: ROFRecords: " << mTracker->getTrackROFRecords().capacity() << " tracks: " << mTracker->getTracks().capacity() << " clusters: " << mTracker->getClusters().capacity();
67  double scaleFactor = 1.e6 / mNROFs;
68  LOG(info) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us tracking: " << mTimerTracker.count() * scaleFactor << " us hitMapBuilder: " << mTimerBuilder.count() << " us";
69  };
70  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
71  }

73  void run(o2::framework::ProcessingContext& pc)
74  {
75  auto tStart = std::chrono::high_resolution_clock::now();
76 
77  auto clusters = pc.inputs().get<gsl::span<Cluster>>("mid_clusters");
78 
79  auto inROFRecords = pc.inputs().get<gsl::span<ROFRecord>>("mid_clusters_rof");
80 
81  auto tAlgoStart = std::chrono::high_resolution_clock::now();
82  mTracker->process(clusters, inROFRecords);
83  mTimerTracker += std::chrono::high_resolution_clock::now() - tAlgoStart;
84 
85  tAlgoStart = std::chrono::high_resolution_clock::now();
86  std::vector<Track> tracks = mTracker->getTracks();
87  mHitMapBuilder->process(tracks, clusters);
88  mTimerBuilder += std::chrono::high_resolution_clock::now() - tAlgoStart;
89 
90  if (mIsMC) {
91  std::unique_ptr<const o2::dataformats::MCTruthContainer<MCClusterLabel>> labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<MCClusterLabel>*>("mid_clusterlabels");
92  mTrackLabeler.process(mTracker->getClusters(), tracks, *labels);
93  pc.outputs().snapshot(of::Output{"MID", "TRACKLABELS", 0, of::Lifetime::Timeframe}, mTrackLabeler.getTracksLabels());
94  LOG(debug) << "Sent " << mTrackLabeler.getTracksLabels().getIndexedSize() << " indexed tracks.";
95  pc.outputs().snapshot(of::Output{"MID", "TRCLUSLABELS", 0, of::Lifetime::Timeframe}, mTrackLabeler.getTrackClustersLabels());
96  LOG(debug) << "Sent " << mTrackLabeler.getTrackClustersLabels().getIndexedSize() << " indexed track clusters.";
97  }
98 
99  pc.outputs().snapshot(of::Output{"MID", "TRACKS", 0, of::Lifetime::Timeframe}, tracks);
100  LOG(debug) << "Sent " << tracks.size() << " tracks.";
101  pc.outputs().snapshot(of::Output{"MID", "TRACKCLUSTERS", 0, of::Lifetime::Timeframe}, mTracker->getClusters());
102  LOG(debug) << "Sent " << mTracker->getClusters().size() << " track clusters.";
103 
104  pc.outputs().snapshot(of::Output{"MID", "TRACKROFS", 0, of::Lifetime::Timeframe}, mTracker->getTrackROFRecords());
105  LOG(debug) << "Sent " << mTracker->getTrackROFRecords().size() << " ROFs.";
106  pc.outputs().snapshot(of::Output{"MID", "TRCLUSROFS", 0, of::Lifetime::Timeframe}, mTracker->getClusterROFRecords());
107  LOG(debug) << "Sent " << mTracker->getClusterROFRecords().size() << " ROFs.";
108 
109  mTimer += std::chrono::high_resolution_clock::now() - tStart;
110  mNROFs += inROFRecords.size();
111  }
112 
113  private:
114  bool mIsMC = false;
115  TrackLabeler mTrackLabeler{};
116  std::unique_ptr<Tracker> mTracker{nullptr};
117  std::unique_ptr<HitMapBuilder> mHitMapBuilder{nullptr};
118  std::chrono::duration<double> mTimer{0};
119  std::chrono::duration<double> mTimerTracker{0};

132  of::OutputSpec{"MID", "TRCLUSROFS"}};
133

136 
137  outputSpecs.emplace_back(of::OutputSpec{"MID", "TRACKLABELS"});
138  outputSpecs.emplace_back(of::OutputSpec{"MID", "TRCLUSLABELS"});
139  }
140 
141  return of::DataProcessorSpec{
142  "MIDTracker",
143  {inputSpecs},
144  {outputSpecs},
145  of::adaptFromTask<o2::mid::TrackerDeviceDPL>(isMC),
146  of::Options{}};
147 }
