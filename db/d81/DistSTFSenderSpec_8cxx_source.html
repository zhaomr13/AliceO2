41 void DistSTFSender::run(ProcessingContext& pc)
42 {
43  const auto* dh = DataRefUtils::getHeader<DataHeader*>(pc.inputs().getFirstValid(true));
44  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(pc.inputs().getFirstValid(true))->creation;
45  LOG(info) << "DIST STF for run " << dh->runNumber << " orbit " << dh->firstTForbit << " creation " << creationTime << " TFid: " << dh->tfCounter;
46  STFHeader stfHeader{dh->tfCounter, dh->firstTForbit, dh->runNumber};
47  pc.outputs().snapshot(o2::framework::Output{gDataOriginFLP, gDataDescriptionDISTSTF, mSubSpec}, stfHeader);
48  if (++mTFCount >= mMaxTF) {
49  pc.services().get<ControlService>().endOfStream();
50  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
51  }
52 }
